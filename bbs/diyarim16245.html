ئىچكىردىكى سىتۇدېنتلار


             ئىچكىردىكى ئۆلكىدە،
             ئالىئىي مەكتەپ بەك تولا.
             ھەربىرىدە ئوقۇيمىز،
            ئۇيغۇر بالىلار ئاز-تولا.
                                                   كەلمەس چاغدا بۇ يەرگە،
                                                    گۈزەل خىيال سۈرگەنتۇق.
                                                    مەكتەپتىكى ھاياتنى،
                                                    بەك پەيزە دەپ ئويلايتۇق.
       
              مانا ھازىر ئېشىپتۇ،
              بىز كەلگىلى ئۈچ يىلدىن.
              سۈرگەن گۈزەل خىيالنى،
              تاپالمىدۇق ھېچ يەردىن.
                                              
                                                   بىر ياتاقتا ياتىمىز،
                                                   ئالتە ئادەم بىر بولۇپ.
                                                   بەشى ئۇلار بىرسى بىز،
                                                   غېرىپلىققا جۆر بولۇپ.

             دەرسلەرگە كىرىدۇ،
             ئاكا مىللەت ئۇستازلار.
              ھەر شىۋىدە سۆزلەيدۇ،
             چۈشەنمەيمىز ئاسانلا.

                                                   لېكىن شۇنداق بولسىمۇ،
                                                   ئىخلاس بىلەن ئاڭلايمىز.
                                                    بەزىسىنى چۈشەنسەك،
                                                    بەزىلىرى جاۋاپسىز.

             مەكتەپلەردە بىردىنلا،
             مۇسۇلمانچە ئاشخانا بار.
             كۈندە كىرسەك ئوخشاشلا،
             گاڭپەن بىلەن موما بار.
        
                                                   سەيلىرىنى سانىساق،
                                                   تۈرى چىقار بەش-ئالتە.
                                                   ئەمما يېسە ئوخشاشتۇر،
                                                   ھەممىسىلا بىر تەمدە.

             لېكىن شۇنداق بولسىمۇ،
             يەيمىز ئۇنى ئامالسىز.
             قاقشىساقمۇ ئاخىرى،
             ئاستا-ئاستا كۆنىمىز.
                                        
                                                   كۈتۈپخانىغا كىرسىڭىز،
                                                    ھەممە ئورۇن توشاقلىق.
                                                    قارىسىڭىز زەن سېلىپ،
                                                    ئۇيغۇر بالىلار ساناقلىق.
             ئۆگىنىدۇ ئەلۋەتتە،
             كۆپ ساندىكى بالىلار.
             لېكىن يوقمۇ ئەمەستۇر،
             ئوينايدىغان بەڭۋاشلار.

                                                    قارىسىڭىز بەزىسى،
                                                    كېچە-كېچىلەپ ئوينايدۇ.
                                                    قوپالمىسا ئەتىسى،
                                                    دەرسلەرگە چىقمايدۇ.

            تېخى بەزى بالىلارنىڭ،
            مۇھەببىتى كاپىردۇر.
            سورىسىڭىز ئۇلاردىن،
             كۆڭۈل خۇشى جاۋاپتۇر.

                                                  بىر چىرايلىق يىگىتلەر،
                                                  بەلدىن تۇتقان ئۇلارنى،
                                                  تېخى گۈزەل قىزلارمۇ،
                                                   قولتۇقلايدۇ ياتلارنى.

          شۇنداقلارنى كۆرگەندە،
          يۈرەك سوقار نەپرەتتە.
          ئۆتكۈزەلمەي ئىدىيىدىن،
           توۋا دەيمىز ھەسرەتتە.
                                                   ئەمما كۆپتۇر يەنىلا،
                                                   ياخشىلىرى كۆپ سانلىق،
                                                   ئىمانىنى بۇلغىغان،
                                                   ئارىمىزدا ساناقلىق.

           بازارلارغا چىققاندا،
           ئۇيغۇر چىراي ئۇچرىماس.
           چۈمىلىدەك ئادەملەر،
           قېشىڭىزدىن ئايرىلماس.
                                                    ئۇچراپ قالسا گاھىدا،
                                                     سالام-سائەت ئاددىلا.
                                                     بەزى تېخى قارىماي،
                                                      ئۆتۈپ كېتىمىز گۇماندا.
          ھېيت ئايەملەر جىممىدە،
           ئۆتۈپ كېتەر ناۋاسىز.
          ئايەم پەيزى سۈرەلمەي،
           يىغلار غېرىپ دىلىمىز.
                                                   لېكىن بىزگە بۇ پۇرسەت،
                                                   ھەرگىز ئاسان كەلمىگەن.
                                                   ھامان ئۇتۇق قازىنار،
                                                    توختىماستىن ئىزدەنگەن.
          شۇڭلاشقىمۇ دوستلىرىم،
           قەدىرلەيلى پۇرسەتنى.
           ھەرگىز يەردە قالمىسۇن،
           ئەلنىڭ كۈتكەن ئۈمىدى.
