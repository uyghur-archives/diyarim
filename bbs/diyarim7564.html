ئۇيغۇرلارنىڭ ئەڭ قەدىمكى ئەجداتلېرى ياكى ئېتنولوگىيىلىك تەركىبلېرى ئىدىل-كاسىپنىڭ شەرقىدىن تارتىپ سېلىنگا بويلېرىغىچە،ئارال-بايقال كۆللېرىنىڭ جەنوبىدىن تارتىپ تاكى قارا قۇرۇم ئىتەكلېرىگىچە بولغان كەڭ زوننىغا تارقىلىپ،ئارىلاش-قۇرىلاش ياشىغانلىقى ئۇچۇن،ئۇيغۇرلارنىڭ ئېتنىك قاتلىمىنى تەشكىل قىلغان ئەلۋان-تۈرلۈك ئۇرۇغ-قەبىلىلەر ھەر زامان ھەر ماكاندا ھەر خىل ئاتىلىپ كەلگەن.
قەدىمكى زامانلاردا،«قەبىلىنىڭ نامى كۇپىنچە تاسادىبى پەيدا بولغان،ئاڭلىق تاللاپ قۇيۇلغان ئەمەس ئىدى.قەبىلىنىڭ ئۆزى ئۆزىگە بەرگەن نامى باشقىچە بولغان ھالدا،قۇشنا قەبىلىلەر تەرپىدىن مەزكور قەبىلىگە بېرىلگەن نام بېرىپ-بېرىم ئۇنىڭ دائىمى نامى بۇلۇپ قالاتتى.بۇنداق ھادىسىلەر كۆپ بۇلۇپ تۇراتتى.»(ف ئىنگىلىس)
«ئۇيغۇر»دىگەن نام قەدىمكى خەنزۇ تارىخ كىتابلېرىدا قەيىت قىلىنغان قەدىمكى نامدۇر.سۇلالىلەرنىڭ ئالمىشىشى ۋە تارىخ تۈزۈگۇچىلەرنىڭ پىكىر-مۇلاھىزىلېرىنىڭ پەرىقلىق بۇلىشى تۇپەيلىدىن،بۇ بىر خىل نام ھەر خىل قەيىت قىلنىپ كەلگەن،پەقەت خەنزۇچە تەرجىمە ناملېرىدىنلا 30خىلدىن ئاشىدۇ.ھەتتا يۈەن سۇلالىسى دەۋرىدىلا 18خىل يېزىلغان.
«ئۇيغۇر»دىگەن نامنىڭ كىلىپ چىقىشى ۋە مەنىسى تۇغرىسىدىمۇ ھەر خىل رىۋايەتلەر ۋە ئاسارەتلەر بار.
1،مەھمۇد قەشقىرىنىڭ«دىۋانۇ لۇغاتىت تۈرۈك»ئەسىرى بۇيىچە؛
«ئۇيغۇر.بىر ئەلنىڭ ئىسمى...زۇلقەرنەيىن ئۇيغۇر ئىلىگە يېقىنلاشقاندا،تۈرۈك خاقانى ئۇنىڭغا قارشى 4000ئادەم ئەۋەتكەن.ئۇلارنىڭ قالپاقلېرىنىڭ قاناتلېرى لاچىن قانتىغا ئوخشايدىكەن.ئوقنى ئالدىغا قانداق ئاتسا،كەينىگىمۇ شۇنداق ئۇستىلىق بىلەن ئاتىدىكەن.زۇلقەرنەيىن بۇلارغا ھەيران قاپتۇ ۋە«ئىنەن خۇز خۇرەنىد»--بۇلار باشقىلارغا مۇھتاج بولماي،ئۆز ئۇزۇقىنى ئۆزى تىپىپ يەيدىغانلار ئىكەن.بۇلارنىڭ قولىدىن ئوۋ قىچىپ قۇتۇلالمايدۇ،قاچان خالىسا،شۇ چاغدا ئېتىپ يىيەلەيدۇ»دەپتۇ.شۇنىڭدىن كىيىن بۇ ئەل «خۇزخۇر»دەپ ئاتىلىپتۇ.كىيىن....«خ»ھەرىپى«ا»غا ئالمىشىش بىلەن،«خۇز»سۆزىدىكى«ز»ھەرپىمۇ«ى»غا ئالماشقان...شۇنىڭدىن كىيىن«خۇز»سۆزىدىكى«خ»ھەرپى «غ»غا ئالماشقان».
«ئۇيغۇر»-خۇزخۇردىن كىلىپ چىققان،مەنىسى»ئۆز ئۇزۇقىنى ئۆزى تىپىپ يەيدىغانلار بۇلۇپ،ماكدۇنلۇق ئىسكەندەر قويغان .بۇ2300يىل ئىلگىرى قۇيۇلغان نام بۇلۇپ چىقىدۇ.
2،خۇجا راشىدىن فەزلۇللاھنىڭ«جامئوت-تەۋارىخ»كىتابى بۇيىچە؛
نوھنىڭ ئوغلى يافەسنىڭ ئوغلى تۈرىكنىڭ ئەۋلادىدىن قارا خاننىڭ ئوغلى ئۇغۇزخان بىردىبىر تەڭرىگە ئىشىنەتتى.شۇ سەۋەپتىن ئۇنىڭ تاغىلېرى،ئاكا-ئۇكىلېرى،جىيەنلېرى ۋە ئىككى تۇغققان بۇرادەرلېرى ئۇتتۇرسىدا جەڭ-جىدەل پەيدا بولدى.بۇ كىشىلەردىن بەزىلېرى ئۇنىڭغا ياردەم بەردى،بەزىلېرى ئۇنىڭ بىلەن جەڭ قىلىپ يىڭىلدى-دە،ئۆز قەبىلىلېرىنى باشلاپ كىلىپ بەيئەت قىلدى.غەلبىدىن كىيىن،ئۇ چوڭ زىياپەت ئۆتكۈزۈپ،ئۆزىنىڭ ئۇرۇغ-تۇققان،سەردار ۋە جەڭچىلېرىدىن ھال سورىدى.ئۆزىگە قۇشۇلغان قەبىلىلەرگە«ئۇيغۇر»دەپ نام بەردى.بۇ نامنىڭ مەنىسى«بىرلىشىش»«ياردەملىشىش»دىگەنلىكتۇر.بۇ نام ھەممە قەبىلىلەرگە قوللىنىپلا قالماستىن،قەبىلىلەرنىڭ تارماقلېرى ۋە بۇلارنىڭ ئەۋلادلېرىغىمۇ قوللىنىلدى.
بۇ باياندىن«ئۇيغۇر»دىگەن نام«بىرلەشكەن»،«ياردەملەشكەن»،«ئۇيۇشقان»دىگەن مەنىنى بىلدۇرۇش بىلەن بىللە،ناھايىتى نۇرغۇن قەبىلىلەرگە ئۇمومى نام بولغانلىقى مەلوم.
3،ئەبۇل غازنىڭ «شەجەرەئى تۈرۈك»كىتابى بۇيىچە؛
تارىخچىلار شۇنداق رىۋايەت قىلغانكى،مۇڭغۇل يۇرتىدا ئىككى تاغ باردۇر.ئۇزۇنلۇقلېرى قۇياش چىقىشتىن قۇياش پېتىشقىچە سوزۇلغان،ناھايىتى ئۇلۇغ تاغلاردۇر.بىرىنىڭ ئىتى«توقراتۇ بوزلۇق»ۋە ئىككىنچىسىنىڭ ئېتى«ئۇسقۇنلۇق تارام».بۇ ئىككى تاغنىڭ ئارىسىدا مۇڭغۇل يۇرتىنىڭ كۈن پېتىشىدا يەنە بىر تاغ باركى،ئېتىنى قۇت تاغ دەرلەر.مەزكور تاغلارنىڭ ئارسىدا بىر يەردە10ساي بار ۋە ئىككىنچى يەردە 9 ساي بار.بۇ سايلارنىڭ بارچىسى ئۇلۇغ سۇلاردۇر.قەدىمكى زاماندا ئۇيغۇر ئىلى شۇ سايلارنىڭ ئارسىدا ئولتۇرار ئىدى.10سايدا ئولتۇرغاننى ئون ئۇيغۇر ۋە 9سايدا ئولتۇرغاننى 9ئۇيغۇر دەرلەر ئىدى.شەھەرلېرى ۋە كەنىتلېرى ھەم ئىكىنزارلېرى كۆپ بۇلۇر ئىدى.120ئۇرۇق ئەل ئىدى.بىر كىشىنى پادىشاھ كۈتۈرۇپ،ئۇنىڭ ھۈكىمىگە ئىتائەت قىلمايتتى.بۇ سەۋەپتىن بىر-بىرى بىلەن ياۋ بۇلۇپ،بۇزۇلۇر ئىدى.بىر كۈنى بارچىسى يىغىلىپ كېڭەشتى ۋە بىز ئىككى تۈرلۈك ئەل ھەر قايسىمىز بىر كىشىنى تۆرە قىلايلى.ھەر كىم ئۇنىڭ سۆزىنى قۇبۇل قىلمىسا،بىشىنى كەستۇرسۇن ۋە مىلىنى ئالدۇرسۇن دىدى.ئون ئۇيغۇر ئۇرۇغىدىن مەنگۈتاي ئاتلىق كىشنى تۆرە كۆتۇرۇپ،ئەل ئىلتەر دەپ لەقەم قويدى.تۇققۇز ئۇيغۇر ئۆز ئۇرۇغىدىن بىر كىشىنى تۆرە كۆتۇرۇپ،كۆل ئەركىن دەپ لەقەم قويدى.بۇ ئىككسىنىڭ ئوغلانلېرى 100يىلغىچە تۆرىلىك قىلدى.ئاندىن كىيىن يەنە بىر ئۇيغۇر بولدى.ئون ئۇيغۇرغا ھەر كىم تۆرە بولسا ئەل ئىلتەر ۋە توققۇز ئەلگە تۆرە بولسا كۆل ئەركىن دىدى.ئۈچ مىڭ يىلغا يىقىن ئۇشبۇ يۇرتتا ئېيتىلغان يۇرتتا ئولتۇردى.ئاندىن كىيىن بۇزۇلدى.ئولجا ۋە ئەسىر بۇلۇپ تۇردى.بىر نەچچىسى يۇرتىدا ئوللتۇرۇپ قالدى ۋە بىر نەچچىسى ئىرىتىش سۈيىنىڭ بۇيىغا كەلدى ھەم 3 بۆلەككە بۈلىندى.
4،«ئۇغۇزنامە»بۇيىچە؛
كۈنلەردىن بىر كۈنى ئايخان كۆز يورۇپ،بۇشاندى.بىر ئۇغۇل تۇغدى.بۇ ئۇغۇلنىڭ ئۆڭى كۆك،ئاغزى چوغدەك قىزىل،كۆزلېرى ھال،چاچلېرى،قاشلېرى قارا ئىدى.كۆنلەرنىڭ بىرىدە(ئۇغۇزخان)ئوۋغا چىقتى.ئۇغۇزخاقان بىر جايدا تەڭرىگە سېغىنۋاتاتتى.(ئالەم)قاراڭغۇلىشىپ،ئاسماندىن بىر كۆك نۇر چۇشتى.(بۇ نۇر)كۈندىن نۇرلۇق،ئايدىن يورۇق ئىدى.ئۇغۇزخان(قىشىغا)بېرىپ،بۇ نۇرنىڭ ئىچىدە بىر قىزنى كۆردى.تەنھا ئولتۇراتتى...........بۇ قىز شۇنچىلىك گۇزەل ئىدىكى(ئۇ)كۈلسە،كۆك تەڭرىمۇ كۈلەر ئىدى،يىغلىسا،كۆك تەڭرىمۇ بىللە يىغلار ئىدى.ئۇغۇزخاقان ئۇ قىزنى كۆردى-دە،ئەس-ھۇشىنى يۇقاتتى ۋە ئۇنىڭغا ئاشىق بولدى......ئۇ قىز ھامىلدار بولدى،كۈنلەر ئۇتۇپ....قىز ئۈچ ئۇغۇل تۇغدى.تۇنجىسىغا كۈن،ئوتتۇرانچىسىغا ئاي ۋە كەنجىسىگە يۇلتۇز دەپ ئات قويدى.
يەنە بىر كۈنى ئۇغۇزخاقان ئوۋغا چىقتى،ئۇ ئالدىدىكى كۆلنىڭ ئوتتۇرسىدا بىر دەرەخ كۆردى.شۇ دەرەخنىڭ كاۋىكىدا بىر قىز تەنھا ئولتۇراتتى.ئۇ بەكمۇ چىرايلىق ئىدى.ئۇنىڭ كۆزى ئاسماندىن بەكرەك كۆك ئىدى.ئۇغۇزخان ئۇنى سۇيۇپ قالدى ۋە ئالدى........ئۇ قىز ئىككى قات بۇلدى....ئۇچ ئۇغۇل تۇغدى.تۇنجىسىغا كۆك(ئاسمان)،ئوتتيۇرانچىسىغا تاغ،كەنجىسىگە دېڭىز دەپ ئات قويدى.شۇنىڭدىن كىيىن ئۇغۇزخان كاتتا توي بەردى...تويدىن كىيىن،ئۇغۇزخان بەگلەرگە،خەلىقكە يارلىق چۇشۇرۇپ مۇنداق دىدى.بىزگە كۆك بۆرە بولغاي ئۇران،كۈن توغ بولغاي،ئاسمان قۇرىغان!
شۇنىڭدىن كىيىن،ئۇغۇزخان 4 ئەتراپىغا يارلىق چۇشۇردى ۋە مەكتۈپ يېزىپ ئەلچىلەرنى ئەۋەتتى.بۇ مەكتوپقا شۇنداق يېزىلغان ئىدى؛«مەن ئۇيغۇرنىڭ خاقانىمەن،مەن پۇتۇن جاھاننىڭمۇ خاقانى بۇلۇشۇم كېرەك........سول تەرەپتە ئۇرۇم دىگەن بىر خاقان بار ئىىدى...يارلىققا بوي سۇنمىدى.ئۇغۇزخان قۇشۇنىنى باشلاپ 40 كۈندىن كىيىن مۇزتاغ دىگەن تاغنىڭ ئىتىكىگە كەلدى ۋە چۇشكۈن قۇرۇپ ئارام ئالدى.تاڭ سۈزۈلگەندە،ئۇغۇزخاننىڭ چىدىرىغا كۈندەك بىر يورۇق چۇشتى.بۇ نۇر ئىچىدىن  كۆك تۈكلۈك،كۆك يايلىق يۇغان بىر ئەركەك بۆرە پەيدا بولدى.بۇ بۆرە ئۇغۇزخانغا گەپ قىلدى؛ھاي-ھاي ئۇغۇز،سەن ئۇرۇن ئۇستىگە ئاتلانماق بوپسەن.ھاي-ھاي ئۇغۇز،ئالدىڭلاردا مەن (يول باشلاپ ماڭىمەن).شۇنىڭدىن كىيىن ئۇغۇزخان..(ئالغا قاراپ )يۇرۇپ كەتتى.قارىغۇدەك بولسا،قۇشۇننىڭ ئالدىدا كۆك تۈكلۈك،كۆك يايلىق چوڭ بىر ئەركەك بۆرە كېتىپ باراتتى....بىر نەچچە كۈندىن كىيىن........بۆرە توختاپ قالدى.ئۇغۇزمۇ قۇشۇننى بىلەن توختىدى.بۇ يەردە ئېتىل دىگەن بىر چوڭ دەريا بار ئىدى.بىر قارا تاغ باغرىدا ئۇرۇش باشلاندى.....ئۇغۇزخان يەڭدى.....شۇنىڭدىن كىيىن،ئۇغۇزخان يەنە ھېلىقى كۆك تۈكلۈك،كۆك يايلىق ئەركەك بۆرىنى كۆردى.ئۇ بۆرە ئۇغۇزخانغا؛«ئەمدى قۇشۇن بىلەن بۇ يەردىن قوزغال،ئۇغۇز!ئادەملېرىڭنى،بەگلېرىڭنى باشلاپ مىڭىۋەر،مەن ساڭا يول باشلاپ ماڭىمەن»دىدى....بىر كۈنى .....ئەركەك بۆرە يەنە يولدىن توختىدى.ئۇغۇزخانمۇ يولدىن توختىدى....شۇنىڭدىن كىيىن ئۇ يەنە شۇ بۆرە بىلەن سىندۇ،تاڭغۇت ھەم شام تەرەپلەرگە يۇرۇش قىلىپ باردى...جەنۇپ تەرەپتە بارقان دىگەن بىر جاي بار ئىدى.ناھايىتى باي مەملىكەت،ئىنتايىن ئىسسىق يەر ئىدى.....خەلىقنىڭ رەڭگىرويى قاپقارا ئىدى.بۇ يەرنىڭ خاقانىنىڭ ئىسىمى ماسار ئىدى....ئۇغۇزخان يەڭدى....ئۇغۇزخاننىڭ يېنىدا ئاق ساقاللىق،مۇز چاچلىق،قابىليەتلىك بىر قېرى كىشى بار ئىدى....ۋەزىر ئىدى.ئۇنىڭ ئىسمى ئۇلۇغ تۈرۈك ئىدى.كۈنلەرنىڭ ب ىرىدە ئۇ چۈشىدە بىر ئالتۇن يا بىلەن ئۈچ كۆمۇش ئوق كۆردى.بۇ ئالتۇن يا كۈنچىقىشتىن كۈنپىتىشقا تارتىلغان،ئۆچ كۈمۈش ئوق بولسا،شىمالنى كۆرسىتيىپ تۇراتتى.(ئۇ).....چۇشىدە كۆرگەنلېرىنى ئۇغۇزخانغا ئېيتىپ بەردى.....ئۇغۇزخاقانغا ئۇلۇغ تۈرۈكنىڭ سۆزلېرى ياقتى....زىمىنىنى ئۇغۇللېرىغا بۈلۈپ بەردى.
بۇ رىۋايەت بۇيىچە ،ئۇغۇزخان پۇتۇن ئاسىيانى دىگۇدەك بويسۇندۇرغان ۋە «كۆك بۆرە»نى سىموۋۇل قىلغان ئۇلۇغ خاقان بۇلۇپ،ئالتە ئوغلىدىن«ئالتۇن يا»غا ئىگە بولغان ئۈچ ئوغلى(كۈن،ئاي،يۇلتۇز)ئوڭ ياققا(برانغارغا)،«كۈمۈش ئوق»قا ئىگە بولغان ئۇچ ئوغلى(كۆك،تاغ،دېڭىز)سول ياققا(جۇۋانغارغا)ھۆكۈمران بولىدۇ.ئۇغۇزخان ئۆزىنى«ئۇيغۇرنىڭ خاقانى»مەن دەپ جاھانغا جاكالايدۇ.

پادىشاھ قىلغانلىقىنى،ئۇنىڭغا ھەممە ئەللەرنىڭ تىلللېرىنى بىلىدىغان ئۇچ قۇزغۇن ئاتا قىلىنغانلىقىنى ۋە ئۇ قاغىلار رازۋىتكا قىلىپ خەۋەر يەتكۈزدىغانلىقىنى،بىر كۈنى كېچىسى چۇشىدە بىر قىزنى كۆرگەنلىكىنى،كىيىن ئۇ قىز بىلەن ئاق تاغدا تاڭ ئاتقۇچە سۆزلەشكەنلىكى،ئاخىرى كېچىسى ئۇ قىز؛«شەرىقتىن غەرىپكىچە بولغان جايلارغا ھۆكۈمران بولىسەن.بۇ ئىشنى ئىجتىھات بىلەن ئورۇنلا،پۇقرالارنى ياخشى باشقۇر»دەپ ۋىدالاشقانلىقىنى،كېيىن يەنە چۇشىدە ئاق كىيىنگەن، ھاسا تۇتقان بىر بۇۋاي بۇ قاغانغا بىر دانە پىستىچە قاشتاشنى بېرىپ«بۇ قاشتاشنى ساقلىيالساڭ،پۇتۇن ئېقلىم سىنىڭ پەرمانىڭ ئاستىدا بۇلىدۇ»دىگەنلىكى ۋە باشقىلار بايان قىلىنىدۇ.
6،مىرخۇنىدنىڭ«رەۋزە تۇس سەفا»دىگەن تارىخ كىتابى؛
قاراخان......ئاتىسى ئۆلگەندىن كىيىن،تەخىتكە ئولتۇرىدۇ.قارا قۇرۇمنىڭ ئارتاق ۋە كەرتاق دىگەن جايلېرىدا يايلاق ۋە قىشلاقلېرى بار ئىدى.......قارا خاننىڭ خۇتۇنى بىر ئۇغۇل تۇغدى.شۇ زاماننىڭ قائىدىسى بۇيىچە ئۇغۇل -قىز بىر ياشقا كىرمىگۇچە ئىسىم قويمايتتى....قاراخان بۇ ئوغۇلغا نىمە دەپ ئات قويساق بۇلار دەپ مەسلىھەت سالغاندا،ئەمدىلا بىر ياشقا كىرگەن بالا،ئىنىق قىلىپ،«مىنىڭ ئىسمىم ئۇغۇز بولسۇن.....»دەيدۇ.شۇنىڭ بىلەن قاراخان ئۆزىنىڭ يېقىن-يورۇقلېرىغا؛«بۇ ئۇغۇل بىزنىڭ ئادىتىمىزنى بۇزدى،ئۇنىڭ ھايات-ماماتلىق تەقدىرى بىزنىڭ ئىلكىمىزدە»دەيدۇ.جاراخان قۇشۇنىنى باشلاپ،ئۆز ئوغلى-ئۇغۇزنى جازالىماق بۇلۇپ ،ئوۋغا چىقىدۇ.....ئۇغۇزخان بۇنىڭدىن  خەۋەر تىپىپ،نائىلاج قارا خان بىلەن جەڭ قىلىدۇ.قاراخان ئەجەللىك زەربە يەپ يىقىلىپ چۇشىدۇ....ئۇغۇزخان تەخىتكە ئولتۇرىدۇ.73يىل خان بۇلىدۇ.
7،ئەرەب-ئەجەم تارىخچىلېرىنىڭ كىتابلېرىدىكى رىۋايەتلەر بۇيىچە؛
جەئىفەر مۇھەممەد بىننى جەرىر تەبەرى«تارىخول-ئۇمەمۋەل -مۈلۈك»دىگەن كىتابەنىڭ تەھمۇرەس ۋە جەمشىد تۇغرىسىدىكى پەسىللېرىدە بايان قىلىشىچە،جەمشىد قېچىپ،زابىلىستانغا بارغاندا،زابىلىستان پادىشاھىنڭ قىزىنى ئالىدۇ.ئۇنىڭدىن تۇغۇلغان ئۇغۇلغا«تۇر»دەپ ئىسىم قۇيىدۇ.تۇردىن بولغان ئۇغۇل دىستان،دىستاندىن بولغان ئۇغۇل فەرامۇز دەپ ئاتىلىدۇ....جەمشىد ئۇرۇغىدىن فەرىدون بۇلۇپ،بۇنىڭ بۇۋىسى بۇنىڭ بۇۋىسى نوھ ئەلەھىسسالامغا ئىمان كەلتۇرگەن.فەرىدۇننىڭ ئۈز ئۇغلى بۇلىدۇ.،چۇڭى تۇر،ئوتتۇرانچىسى سەلىم،كەنجىسى ئەيرەج.........فەرىدۇن جاھاننى پەرزەنىتلېرىگە تەقسىم قىلغاندا،تۈرۈك،خەرەج ۋە ماچىن ھەم مەشرىق زەمىنىنى تۇرغا بېرىدۇ ۋە ئۇنىڭغا فەغفۇز(تەڭرى ئوغلى)دەپ ئات قۇيىدۇ.
دېمەك بۇ رىۋايەت بۇيىچە،ئوتتۇرا ئاسىيا ۋە شەرىققە تارالغان قۇۋۇملار«تۇر»نىڭ ئەۋلادى بۇلۇپ ھىساپلىنىدۇ.
8،«قۇتلانچىن ھەزرەتنىڭ ئابىدىسى»(خەنزۇچە)بۇيىچە؛
كەمىنە خۇاڭ جىن ئېھتىرام بىلەن ئىلاۋە قىلدۇكى،جامائەت تەۋسىيە قىلغان قۇتلانچىن ئۇيغۇرلاردىندۇر.ئۇزاق بۇۋىسى مەزكور ئەلنىڭ پادىشاھىدۇر.شۇ ئەلدە ئىككى دەرەخنىڭ جۇپلىشىشدىن پورەك ھاسىل بولغان ،پورەك يېرىلىپ،بەش ئۇغۇل پەيدا بولغان،ئۇلاردى تۆتى ئۈلۈپ،كەنجىسى يالغۇز قالغان،بۇنى خاسىيەتلىك بىلىپ،ئىززەتلىگەن،ئۇنىڭغا خۇتۇن ئىلىپ بەرگەن.ئۇنىڭ ئەۋلادى ئەلگە پادىشالىق قىلغان.
9،قوچۇ پادىشاھلېرىنىڭ شەجەرە تاش پۇتۇكى(خەنزۇچە)بۇيىچە؛
ئۇيغۇرلارنىڭ زەمىنىدە قارا قۇرۇم دىگەن تاغ بۇلۇپ،ئۇنىڭدىن ئىككى دەريا ئىقىپ چىقاتتى.بىرىنى تۇغلا،يەنە بىرىنى سېلىنگا دەيتتى.كۈنلەردىن بىر كۈنى ئىككى دەريا ئارلىقىدىكى دەرەخكە ئاسماندىن نۇر چۇشۇپتۇ.بۇ ئەلدىكىلەر بۇنى ئاسراپتۇ.دەرەختىن پۇقاق ھاسىل بۇلۇپ،خۇددى ئادەمدەك تولغاق يەپتۇ.نۇر داۋاملىق چۇشىۋېرىپتۇ.9ئاي 10كۈن بولغاندا پۇقاق يېرىلىپتۇ.بەش ئۇغۇل ئاپىرىدە بۇپتۇ.بۇلارنى بىقىپتۇ.بۇلاردىن كەنجىسى بۈگۈ قاغان ئىكەن.بۇ چوڭ بۇلۇپ خەلىقكە باش بۇلۇپ،ئىتىز-ئېرىق ئىشلېرى بىلەن مەشغۇل بۇپتۇ.پادىشاھ بۇپتۇ.بۇنىڭ ئەۋلادىدىن 30پادىشاھ ئۆتۇپتۇ.يۈرۈن تېكىن ۋاقتىغا كەلگەندە،تاڭ خانلىقى بىلەن كۆپ قىتىم ئۇرۇش پەيدا بۇپتۇ.
10،«گاۋجۈ»تەزكىرىسى بۇيىچە؛
«ئادەتتە»ئېيتىلىشىچە،ھۇن تەڭرىقۇتىنىڭ2قىزى بار ئىدى.چىراي-تۇرقى ناھايىتى گۈزەل ئىدى.قۇۋمى بۇ قىزلارنى خاسىيەتلىك دەپ بىلەتتى.تەڭرىقۇت؛«بۇ قىزلېرىمنى ئۆز لايىقىغا بېرىش ئۇچۇن كۆك تەڭرىسىنىڭ ئىختىيارىغا قۇيىمەن»دەپتۇ.شۇنىڭ بىلەن ئەلنىڭ شىمالىدا ئادەم يوق جايغا ئىگىز تۇرا ياساپتۇ.2 قىزنى شۇ يەرگە چىقىرىپ،«كۆك تەڭرىگە تاپشۇردۇق»دەپتۇ.3 يىل ئۆتكەندىن كىيىن،قىزلارنىڭ ئانىسى سېغىنىپ كىتىپ ياندۇرۇپ كەلمەكچى بوپتۇ.تەڭرىقۇت«بولمايدۇ،ياندۇرۇپ كىلش ۋاقتى ئەمەس دەپتۇ».يەنە بىر يىل ئۆتۇپتۇ.بىر قېرى بۆرە كېچە-كۇندۇز تۇرانى ساقلاپ ھۈۋلاپتۇ.تۇرانىڭ تۇۋىدە ئورا بار ئىكەن.كىرگىلى بولمايدىكەن.كىچىك قىز؛ئاتام بىزنى بۇ يەردە كۆك تەڭرىگە تاپشۇرغان،ئەمدى بۆرە كەپتۇ،بەلكىم بۇ خىسلەتلىك مەخلۇقتۇر،كۆك تەڭرى ئەۋەتكەن بولسا كېرەك،دەپ چۇشمەكچى بۇپتۇ.ئاچىسى ھەيران بۇلۇپ،»بۇ-ھايران تۇرسا،ئاتا-ئانامغا نۇمۇس بۇلىدۇ!»دىسىمۇ،سىڭلىسى ئۇنىماي،چۇۇشپ،بۆرىگە تىگىپتۇ.ئۇنىڭدىن تۇغۇپتۇ.بارا-بارا ئاۋۇپ ئەل بۇپتۇ.
11،«يۆياڭ تەرمىلېرى»بۇيىچە؛
«شىمالى تۇتقۇنلارنىڭ سوق ئىلىدە نىپ-شىر ئۇلۇسى بار ئىدى،ئىككىنچى خۇتۇن 4 ئۇغۇل تۇغۇپ،بىرى ياۋا غازغا ئايلىنىپ كەتكەن،قالغان ئۈچ ئۇغۇل بىر-بىرىگە؛گۈزە-گۈزە دەپتۇ.بۇ »ئۆكۈز«دىگىنى ئىكەن.ئۇچ ئۇغۇل كالىغا ئەگىشىپتۇ،كالىنىڭ سۈتى ۋە ئېتىدىن ئۇزۇقلىنىپتۇ».
يۇقۇرىدا نەقىل قىلىنغان رىۋايەت ياكى ئاساتىرلارنىڭ بەزىلېرىدە ئۇيغۇرلارنىڭ ناھايىتى قەدىمكى زاماندىكى يەنى ئانىلىق ئۇرۇقداشلىق جەمئىيىتى ۋاقتىدا پەيدا بولغان ئەقىدە-چۇشەنچىلېرى بۇيىچە ياكى قۇشنا ئەللەر ئىچىدە تارقالغان(ئۇيغۇرلار تۇغرىسىدىكى)ئەپسانىلار بۇيىچە،ئۇيغۇرلارنىڭ كىلىپ چىقىشى ئىلاھىيلاشتۇرۇلۈپ ياكى ئۇزاق ئۆتمۈشتىكى توتېملارغا باغلانغان بايانلاردۇر.بۇ رىۋايەتلەردىن بەزىلېرى بىر ئاتىلىق(پرودىدا)نەزەريىسىگە ئاساسلانغان بۇلۇپ،بىر يىلتىزدىن تارالغان دىگەن پىكىرنى ئالغا سۈرىدۇ.
