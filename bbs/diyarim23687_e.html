چوڭ كىچىكلىكى :كىچىك نورمال چوڭ

 ياخشى كىلىن بولۇشنىڭ 10چوڭ پىرىنسىپى

< قانداق قىلغاندا ياخشى كلېن بولغىلى بولىدۇ ؟> ناملىق سۆھبەتكە قىسقا جاۋاپ:
1،ياخشى ئائىلە تەربىيەسىكۆرگەن ۋە مەڭگۈ ئۆزىنى تەربىلەپ تۇرۇشقا ماھىر بولۇپ،قەلبى ھۆر دىلى پاك بولۇشقا تىرىشىش لازىم.ئەقىللىق مەسئۇلىيەتچان بىر كىلىن كىرگەن ئائىلە خۇددى بۇ دۇنيانىڭ جەننىتىگە،ئەكسىنچە بولغان ئائىلە بۇ دۇنيانىڭ دوۋزىقىغا ئوخشايدۇ؛
2،ھەر ۋاقىت ئۆزىنى قەدىرلەش،ئەخلاقى -پەزىلەتتە نوخسانسىز بولۇش، ئەخلاق-پەزىلەت مەسىلىسىدە شەرتسىز ھالدا  ئۆزىگە دائىم يۇقىرى ئۆلچەم قاتتىق تەلەپ قويۇش لازىم.ئەخلاق-پەزىلەتتە نۇخسان بار  كىلىننى ھەرقانداق بىر قىيىن ئانا-قىيىن ئاتا ۋە ياكى ئەر ئەسلا _مەڭگۈ ياقتۇرمايدۇ ۋە كۆزىگە ئىلمايدۇ؛
3،ئەقىل پاراسەتلىك بولۇش،چۈنكى پاراسەتلىك بىر ئايال ھەر قانداق مۇرەككەپ زىدىيەتلىك مەسىلەرنى ئۈنىملىك كۆڭۈلدىكىدەك ھەل قىلىپ كىتەلەيدۇ.رىئال تۇرمۇشتا تالايلىغان مەسىلىلەر كىچىككىنە شەخسى ھىسيات ياكى قارا كۈچكە تايىنىپ كالتا پەملىك قىلىش بىلەن ئەمەس،بەلكى ئەقىل پاراسەت بىلەن ئاندىن ياخشى ھەل بولىدۇ؛
4،كەڭ قورساق،ئالىجاناپ،ئەپۇچان بولۇش،باشقىلارنى جۈملىدىن كىلىن بوپ كىرگەن شۇ ئائىلىدىكى ھەممى كىشىنى چۈشۈنىشكە ماھىر بولۇش،ۇلارنىڭ روھى دۇنياسى ۋە كىشىلىك قىممەت قاراشلىرىنى  چۈشۈنىش ۋە ئاندىن چوقۇم «ئەل رايى» بويىچە ئىش كۆرۈش ۋە يوتقانغا قاراپ پۇت سۇنۇش كىرەك،ئۆزنى ھەرگۈر تۆۋەن ساپالىق،پەكەش رەزىل، روھى زەيىپ، روھى چۈشكىن، پەزىلەتسىز ئاياللارغا سىلىشتۇرماستىن،نەسلى ئېسىل ،ئەتراپلىق تەربىيە كۆرگەن،ئىستىداتلىق ،مۇنەۋۋەر ئاياللارغا سىلىشتۇرۇپ تۇرۇش، ئۆزىدە بىرەر كەمچىلىكنى ھىس قىلغان ھامان تۈزىتىشن،باشقىلاردا بىرەر ئارتۇقچىلىقلارنى سەزگەن ھامان ئۈگۈنىش كىرەك؛
5،ئاغزى گاھال،قۇلىقى يۇمشاق،پۇت قولى چاپسان،ئىشچان تىرىشچان،پاكىزە بولۇشئەدەپ يوسۇنلۇق بولۇش، ئادىل ھەققانى  ئىش   قىلىش ،بىراق مۇۋاپىق يوسۇندابولۇش ئادىللىق ھەققانىلىقنى ئىش قىلىمەن دەپ قىيىنئانىسىنى رەنجىتىپ قويماسلىق؛،خۇدانىڭ بەندىلەرنىى كۆپ ئاڭلاپ  ئاز راق سۆزلىسۇن دەپ ئاتايىتەن قۇلاقتىن2ئىككىنى،ئېغىزدىن بىرنى ياراتقانلىقىنى ئەستىن چىقارماسلىق،مەمەندانلىق قىلماسلىق ،تاپا-تەنە  پىتنە پاساتتىن نېرى تۇرۇش لازىم ؛
6،تۇرمۇشقا جىددى ۋە ئەستايىدىل مۇئامىلە قىلىش،ئائىلىسىگە سەمىمى -ساداقەتمەن ،مەسئۇلىيەتچان بولۇش؛كەمتەر، ئىغىر بىسىق،مۇلايىم،سېلىق -سىپايە بولۇش،ئەر ئايال مۇناسىۋەتلىرىنى بىر تەرەپ قىلىشقا ماھىر بولۇش،ئۆز ئېرى ئالدىدا ئۆزىنىىڭ ھىسياتىنى يوشۇرۇپ باشقا ئەرلەرگە بولسا«غىلجىڭشىپ»،سۈيىنى چىقىرىدىغان ھاماقەت پەسكەس روھى خائىشلاردىن ھەزەر ئەيلەش ، ئەتىدىن كەچكىچە ئېرىنى كۈللەپ جىنىغا كىرىشىۋىلىپ ئەرنى بىزار قىلدۇرغاندىن كۆرە ئەقىل ئىشلىتىپ ئۆزىنىڭ خوتۇنلۇق بۇرچى ۋە مەجبۇريەتلىرىنى نەمۇنىلىك بىلەن ئادا قىلىپ ۋە ئېرىنىڭ تۇرمۇشىنى رەڭدار مول مەزمۇنلۇق،ئەھمىيەتلىك- ياخشى ئورۇنلاشتۇرۇپ، ۋاختىنى ئۈنىملىك باشقۇرۇپ ۋە كانتىرول قىلىپ، «يولدىن چىقىپ كىتىش»نىڭ ئالدىنى ئېلىش لازىم. قىزلار شۇنى ئىسىدە مەھكەم تۇتىشى كىرەككى: ھەر قانداق ئەر ئەسلىدە ئەسكى بۇزۇق شاللاق بولغانلىقىدىن ئەمەس بەلكى  ئۆز تۇرمىشىدا بىرەر كەمتۈكلىكلەرنى ھىس قىلغاندىن كىيىن ،تۇرمۇشىدىكى ئەنە شۇ كەم تەرپلەرنى تولدۇرۇش ئۈچۈن ئاندىن «باشقا ياقلار»غا ئېزىپ- تېزىپ قالىدۇ؛ئەقلى بار بىر ئايال ئۆزىنىڭ گىرىمچىكلىكى ،ئازازۇللىقى،بىشەم-رەسۋالىقى بىلەن ياكى ئەرنى كۈللەش،قولىغا پالاق ئىلىپ« چىۋىن قورۇغاندەك» قورۇپ يۈرۈش ئارقىلىق بىر ئەرنى ئۆزىگە مەڭگۈ تارتالمايدۇ ھەم قارىتالمايدۇ،ئەكسىچىە يۇقارقىدەك ئەقىللىقلىقى بىلەن  يولۋاستەك كۈچلۇق -ياۋۇز بىر ئەرنىمۇ قوينىڭ قوزىسىدەك يوۋاش، پىلە قۇرتىدەك يۇمشاق قىلىپ، ئۆز ئەتراپىدىن پەرۋانىدەك چۆگۈلەپ كىتەلمەيدىغان،«سىزىقى»دىن چىقمايدىغان،ئۆز خوتۇنىدىن باشقا ئاياللار ئۇ ئەر ئۈچۈن ئۆز مەۋجۇتلىقى ۋە خاسىيىتىنى يوقاتقاندەك ھالەتكە كەلتۈرۋىتەلەيدۇ؛
7،تىرىشىپ تۈرلىك مۈرەككەپ زىدىيەتلىك مەسىلىلەرنى ھەك قىلىشقا ماھىر بولۇش؛تۇرمۇش ۋە ئائىلە ئىشلىرىدا تىجەشلىك رەتلىك تەرتىلىك بولۇش،ئۆزىنى تۈزەشتۈرۈپ يۈرۈش،ئۆز ئائلىسىنىڭ جەننەت كەبى ئىللىق بىر ئائىلە ئىكەنلىكىنى ھە ۋاقىت ھىس قىلدۇرۇپ تۇرۇش،  چاكىنا تەنتەكلىك  سۆلەتۋازلىق مودا كەشلىك قىلماسلىق،ەقىلسىزلىك قىلماسلىق ، گىرىمنى باشقىلارغا كۆز-كۆز قىلىش ئۈچۈن  ياكى باشقا يات جىنىستىكىلەرنىڭ غىدىقىنى كەلتۈرۈش ۋە ياكى باشقىلارنى جىسنى جەلىپ قىلىش ئۈچۈن ئەمەس،  پەقەت ۋە ئۆز ئېرىگە ئۆز خوتۇنىنىڭ تەڭداشسىز  گۈزەل ۋە ئۆزگىچە خاسلىققا ئىگە ئوماق ھەم جەلىپكار ئىكەنلىنى ھىس قىلدۇرۇپ «تۈلكە جىن»لارنىڭ ئازدۇرۇپ كىتىشىدىن  ئىدىۋىي مۇداپىيە كۆرۈش ۋە باشقىلارغا بولسا بۇ خانىم تەربىيە كۆرگەن ئۆزىنى قەدىرلەيدىغان ،تۇرمۇشى تەرتىلىك خاىنم ئىكەندە دەيدىغان تۇيقۇ بىرىش ئۈچۈنلا بولۇشى.سىرتقا ماڭغاندا «چىمەنتاغنىڭ مايمۇنى»دەك ياسىنىپ،ئۆي ئىچىدە بولسا«ئۆي خىزمەتچىسى»دەك پەلى-پەتىش يۈرۈشتىن  ھەزەر ئەيلەش  كىرەك؛
8،ئەستايىدىل ،سەمىمى بولۇش،بەلگۈلىك قۇربان بىرىش،ئۆزىنى بىغىشلاش روھىغا ئىگە بولۇش؛
غەيۋەتكورلۇق ،سۈخەينچىلىك ،ھەسەتخورلۇق ،پىتنە-پاساتتىن گەپ توشۇشتەك ناچار ئىللەتلەرىن ھەزەر ئەيلىشى،ئائىلە ئىچىدىكى چوڭ كىچىك ئىشلاردا پىرىنسىپ جەھەتتىن  قىيىن ئانا-قىين ئاتىسىنىڭ مەيدانىنى قوللشا ۋە شۇلار تەرپتە تۇرۇش،ئائىلە سىرتىدىكى ئىشلاردا شەرتسىز ھالدا ئۆز ئېرىنىڭ مەيدانىدا تۇرۇش ۋە ئۇنى قوللاش، تۇرمۇشىنى باشقىلارنىڭ پىكىرى ۋە ئارزۇسى بويىچە ئەمەس ئۆزىنىڭ ئارزۇسى بويىچە تۇتۇش،باشقىلار ئۈچۈن ئەمەس، ئۆزى -ئۆز ئائىلىسى ئۈچۈن ياشىشى كىرەك؛
9،  ئاتا-ئانىسى ۋە قىيىنئانا،قىيىنئاتىسى ۋە  ئالاقىدارلا ئالدىدا ھەر ۋاقىت ئۆز ئېرىنىڭ ياخشى تەرىپىنى قىلىپ تۇرۇش،ئەرنىڭ ئەيىبى بولسا قۇلىقىغا  ئۇتىقى بولسا ئاتا ئانىسىغا بۈلدۈرۈپ تۇرۇش،ئېرىگە بولغان مۇھەببىتىنى دائىم قېينىئانىسىغا بىلدۈرۈپ تۇرۇش بىلەن بىللە ، قېينىئانىسىنى چوڭ كۆرۈدىغانلىقىنى ياخشى  كۆرىدىغانلىقىنى ھىس قىلدۇرۇشى، قېينىئانىسىغا ئۇنىڭ تۇرمۇشىدىكى مۇھىم ئورنىنى،چوقۇم تىرىشىپ بىر ياخشى كىلىن،  سادەقەتمەن- ۋاپادار خوتۇن مىھرىبان ئانا بولۇپ، چوڭلارنى خاتىرجەم ئۆزى ۋە ئېرىنى ھەقىقى بەخىتلىك قىلىش ئارزۇسىنىڭ  ئىنتايىن كۈچلۇق ئىكەنلىكىنى بىلدۈرۈپ قويۇشى ۋە بۇ ھەقتە پات-پات تەلىم بىرىپ  تۇرۇشىنى تەلەپ قىلىپ تۇرۇش كېرەك ؛
10،كىلىن بولۇپ كىرگەن ئائىلىدىكى بارلىق   ئائىلە ئەزالىرىنى ئەسلى ئۆز ئائىلىسىدىكى قىرىنداشلىرىدەك يىقىن كۆرۈش ۋە ھۆمەرلەش،كۆيۈنىش،ھەرقانداق ئەھۋالدا  ئېرى ھەم قىين ئاتا ،قىيىن ئانىلىرىنى ھۆرمەتلەش، ئىززەت قىلىش ئۇلارنىڭ ئىناۋىتى ۋە غورۇرىغا ھۆرمەت قىلىش،ئۆزىگە راۋا كۆرمىگەن ھەرقانداق ئىشنىباشقىلارغىمۇ راۋا كۆرمەسلىكلازىم.
                خاتىمە:ئەلۋەتتە ،ياخشى كىلىن بولۇش بۇ  ئىنياتىن مۇشەققەتلىك بولغان مۇرەككەپ بىر سېستىما قۇرلىشى بولۇپ ،ئۇنىڭ  قانداقتۇر مەيەييەن بىر قىلىپ ۋە ئەندىزىسىمۇ ھەم يوق.بىراق مەن شۇنىڭغا قەتئى ئىشىنىمەنكى، ياخشى بىر كىلىن  ۋە بەخىت- تەلەييلىك بىر ئايال بولۇپ ،ئۆزنىڭ ئىززەت ئىكرامى بىلەن ياشاشنى ئارزۇ  قىلىدىغان ھەر قانداق بىر قورامىغا يەتكەن قىز(كىلىنلەر) يۇقارقى 10پىرىنسىپ ۋە تەلەپكەئاڭلىق رىئايە قىلىپ ۋە ئۇنى ئۆز ھەركىتىنىڭ قىلبىنامىسى قىلىش بىلەن ،تىرىشىپ ئۆز ئەمىلىيىتىدە كۆرسەتىسىلا   جەزمەن  ئۆز ئارزۇسىغا يىقىنلىشالايدۇ ۋە ھامان يىتەلەيدۇ.جانابى ئاللاھ شۇنداق قىزلار ۋە كىلىنلەرنى مۇرادىغا يەتكۈزگەي~!شۇنداق قىلالىغان ئاياللار ھەم مەڭگۈ ئۆز ئەرلىرىنىڭ چەقىقى مېھرى-مۇھەببىتىگە مۇيەسسەر بولغاي~!
