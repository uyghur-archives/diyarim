      [p][/p]                                      تەڭسىزلىك

                                                  (1)

                                            ئۆي،ئۆيلەردە
                                            كۆيەر لاۋۇلداپ،
                                            تارىمنىڭ گېزى_
                                            يىراق شەرىقتە.
                                            قوتۇر ئېشەك،
                                            ئەبجەق ھارۋىدا،
                                            سوك-سوك تېرىپ،
                                            يۇرىدۇ ئۇيغۇر
                                            يىقېن غەرىپتە.

                                                  (2)

                                            چاپىدۇ ئۆستەڭ،
                                            بىچارە دېھقان،
                                            يىقىلىپ قوپۇپ،
                                            ھەتتا ئۆمىلەپ.
                                            سۇ كەلگەندە-
                                            تۇغان بىشىدا،
                                            كۆك بېشى تەييار:
                                            «سۇيىڭنى تۇتۇش،
                                              پۇلۇڭنى تۆلەپ»!

                                                      (3)

                                            مازاق قىلىپ كۈلدى بىرەيلەن،
                                            يۈگۈرگەندە ئۆردەكنىڭ،
                                            ئۆز بەدىنىنى،
                                            ئىككى ياققا ئۇرغانلىغىنى.
                                            شۇنداق مەستخۇش كۈلكىلەر ئىچرە،
                                            ئۇنتىدى ئۇ،ئۆزىنىڭ_
                                            ھاسا تۇتۇپ تۇرغانلىغىنى.
