بۇ يازمىنى ajrim ﺧﻪﯞﻩﺭﻟﻪﺭدىن بۇ سەھىپىگە كۆچۈرۈپ كەلدى(2008-06-10)
ئابدۇقادىر جالالىدىن ئەپەندى !سىز زادى تەتقىقاتچىمۇ ؟ياكى < پەيلاسوپ> مۇ ؟`~~~سىز قايسى خىل ئاتاشنى مۇۋاپىق دەپ قارايسىز !جەمىيىتىمىزدە سىزنى پەيلاسوپ دەپ قارايدىغانلارمۇ بار ،
تەتقىقاتچى دەپ قارايدىغانلارمۇ بار ..ھەتتا بەزى نوپۇزلۇق ژورناللىرىمىز سىزنى < زامانىمىزنىڭ ياش پەيلاسوپى > دەپ مۇقاۋىغىمۇ چىقاردى ،،،،سىزنىڭچە بۇ بەك مۇبالىغە بولۇپ كەتتىمۇ قانداق .......سىز ئۆزىڭىزنى پەيلاسوپلۇق مۇقامىغا يەتتىم دەپ قارامسىز ؟! ( [بۇ سۇئالغا سىز مىنىڭ خەلقىم مىنىڭ ھەققىدە ناھايىتى ياخشى باھا بېرەلەيدۇ !!!دېگەندەك جاۋاپ بەرمەسلىكىڭىزنى ئۈمۈد قىلىمەن ) مەن سىزنىڭ مائارىپ جەھەتتىكى تۆھپىللىرىڭىزنى بىلىمەن، ھەم بۇ جەھەتتىكى تەتقىقاتىڭز ھەققىدە كۆپ تەرىپلەشلەرنى ئاڭلىغان ئەمما سىزنىڭ ئۆزىڭىزگە خاس ئىدىئولوگىيە سېستىمىڭىزنىڭ بارلىقىنى بىلمەيدىكەنمەن .( بۇ بەلكىم مىنىڭ سىز سۆزلىگەن ،بىزگە تونۇشتۇرۇپ بەرگەن چەتئەل پەيلاسوپلىرىنىڭ كىتابلىرىنى كۆپرەك ئوقۇمىغانلىقىمدىن بولسا كېرەك ) مىنىڭچە سىزنىڭ لىكسىيەلىرىڭىزنى ئاڭلىغان ھەربىر سىتودىنت سىزنىڭ پەلسەپىۋى كۆزقاراشلىرىڭىزنى تولۇق چۈشىنىمەن دىيەلمەيدۇ .چۈنكى كۆپ قىسىم  سىتودىنتلىرىمىزنىڭ سەۋيىسىنىڭ ،بىلىم قۇرۇلمىسىنىڭ ، چۈشۈنۈش ئىقتىدارىنىڭ قانچىلىك ئىكەنلىكى سىزگە ئوخشاش ئالى مەكتەپ ئوقۇتقۇچىسىغا ئايان ئەلۋەتتە .سىز شۇنچىلىك ( ناھايىتى تار ~~ناھايىتى تار )چۈشۈنۈش ئىقتىدارىغرا ئىگە بىر قىسىم كىشىللىرىمىزنىڭ سىزنى پەيلاسوپ دەپ ئاتىشىنى قوبۇل قىلالامسىز ؟!سىزنى باشقىلار، خىزمەتداشلىرىڭىز ،دوسلىرىڭىز ،ئوقۇغۇچىللىرىڭىز ۋە  باشقىلار، پەيلاسوپ دەپ ئاتىغاندا سىز قانداق ھىسياتتا بولىسىز ؟!ئۆزىڭىزنى شۇ خىل ئاتاشقا لايىق كۆرەمسىز ؟!
مىنڭچە سىز < پەيلاسوپ > نىڭ قانداق كىشى  ئىكەنلىكىنى مەندىن ياخشى بىلىسىز .سىز ئوقۇغۇچىللىرىڭىزغا بۇ توغرىسىدا ئېنىقلىما بەردىڭىزمۇ ؟!
سىزنى بىرقىسىم كىشىلەر < پەيلاسوپ > دەپ ئاتىغاندىن كىيىن سىز ئوتتۇرغا چىقىپ ئىنىق بىر چۈشەنچە بەرمىدىڭىز ، ھەتتا بۇ خىل خاتالىقنى كۆرمەسكە سېلىۋىلىش پوزىتسىيىسىدە بولۇۋاتىسىز .
ئەمما بىزگە ئوخشاش سىزنىڭ مائارىپ تەتقىقاتىڭىزنى قوللىغۇچىللىرىڭىز سىزنىڭ خەلقىمىز ئارسىدىكى يامان ئادەتلەرنىڭ بىرى بولغان خوشامەتچىلىكنىڭ قۇربانىغا ئايلىنىپ قېلىشىڭىزدىن كۆپ ئەنسىرەيمىز ...( تەڭرىم سىزگە ماغدۇر ~ئىدراك بەرگەي )...........

كىتاپخانلارنىڭ سەمىگە سىلىپ قۇيۇش كېرەككى،مەن پەيلاسوپ ياكى شائىېر بۇلۇش ئۈچۈن يازمىدىم.مەن ئىنسانى قەدىر-قىممىتىمنى ئاشۇ ناملار تۈپەيلى چاكىنىلاشتۇرۇپ قۇيۇشنى خالىمايمەن.ئىنسانلار ئالتۇننى ئۆزى ياخشى كۆرگەنلىكى تۈپەيلى قۇشلارنىڭ قانتىغا زىننەت قىلىپ ئىسىپ قويسا،قۇشلار ئەركىنلىكتىن مەھروم بۇلىدۇ.مەن ئىنسان،ماڭا پەقەت مۇشۇ شان-شەرەپلا يېتىپ ئاشىدۇ.
