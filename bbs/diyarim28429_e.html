ئاپا، مەن ھېيىتلىق كىيىم كىيمەيمەن

   



    قۇربان ھېيىتنىڭ يېقىنلىشىشىغا ئەگىشىپ بازار،دۇكانلاردا ئايلىنىۋاتقان ئادەملەرمۇ كۆپىيىپ ھەممە ياقنى ناھايىتى قىزغىن،ئالدىراش بىر خىل كەيپىيات قاپلىغان ئىدى.شۇنىڭدەك ئىدارىمىزدىكى خانىم -قىزلارمۇقوللىرىدىكى ئىشلارنى ھاپىلا-شاپىلا تۈگىتىپ پۇرسەت بولسىلا ئۇ،بۇ ئىشخانىلارغا يىغىلىۋىلىپ ھېيىتنىڭ تەيارلىقلىرى،بازار چۆگىلەش،كىيىم سېتىۋىلىش،قايسى دۇكانغا قانداق يېڭى مال كەلگەنلىگى،ئۆزلىرىنىڭ نىمەئالغانلىغى توغرىسىدا بەس-بەس بىلەن پاراڭلىشىپ ۋاقىت ئۆتكىزەتتى.تېخى بەزىدە بۇ پاراڭلارغا يېنىدىن ئەتىر ئايرىماي،ساقال-بۇرۇتلىرىنى پاكىزە قىرىپ،ئايىغىنى پارقىرىتىپ،شىملىرىنىڭ قىرلىق ھالىتىنى بۇزمىغان ھالدا كىيىپ ياسىنىشقا بىر خىل ئېتىۋار بىلەن قارايدىغان ئەر خىزمەتداشلىرىمىزمۇ ئارىلىشىپ قالاتتى.
   بۇگۇن ئەتىگەن ئىشقا كېلىپ ئىشخانىنى تازىلاپ بولۇپ كومپىيۇتۇرنىڭ ئالدىدا ئەمدىلا ئولتۇرۇشۇمغا ياندىكى ئىشخانىدا ئىشلەيدىغان ئۆزى  دۇغىلاق كەلگەن، سارغۇچ بوياپ،بۈدرە قىلىپ كۆپتۇرۇۋالغان چاچلىرى تولۇق كەلگەن يۈزىگە ماسلاشقان،ئوتتۇرا ياشلىقىغا قارىماي قاتار چاي ئويناۋاتقان ئاياللىرىنى«بىزنىڭ قىزلار»دەپ سۆزلەيدىغان،پومساق بارماقلىرىدىكى ئۈزۈكلىرى كۈندە دىگۈدەك ئالمىشىپ تۇرۇدىغان ئاسىيەم ھەدە كىرىپ كەلدى.
  _قانداقراق ئەھۋالىڭىز ئۇكام؟
  _ياخشى ئاسىيە ھەدە،ئۆزىڭىزچۇ؟
  _خۇداغا شۈكرى_ئۇ سوزۇنچاق يېشىل كۆز قويۇلغان ئۈزۈكىنى ۋال-ۋۇل قىلىپ بارماقلىرىنى قارسىلداتقىنىچە جاۋاپ بەردى-دە،تەكەللۇپسىز ياندىكى ئورۇندۇققا جايلاشتى.
  _ۋاي ئاللا،ئەتىگەندە كېلىپلا كومپىيۇتۇرغا چاپلاشتىڭىزمۇ ئەندى؟
  _ھە،تۈنۈگۈن بىر ماتىرىيالنىڭ ئازراق چالىسى قالغان ،شۇنى قولۇمدىن چىقىرىۋىتەي دەپ.
  _ۋاي-ۋۇي،كىچىككىنە تۇرۇپ خىزمەت،خىزمەت دەپلا ئۆتكەن،ئەتىگەندە بىر كەلسىڭىز كىرىپلا كېتىسىز،سىزنى بەزىدە ھەپتە،ھەپتە كۆرگىلىمۇ بولمايدۇ_ھە؟ھى ھى ھى خىزمەتداش بولغاندىن كېيىن ئانچە-مۇنچە ئارىلىشىپ ئۆتمەمسىز دەيمەن،قاراڭ سېغىنىپمۇ قالىدىكەن ئادەم سىزنى،
  _شۇنداقمۇ؟مەن،شۇ،ئادەت بولۇپ قاپتۇ ھەدە_مەن ئوڭايسىزلانغان ھالدا ئۇنىڭغا قاراپ جىلمىيىپ قويدۇم.
  _كۆڭلۈم سىزنى تارتىپ پاراڭغا كىردىم قاراڭ،ئالدىراش بولسىڭىز ئىشىڭىزنى قىلىۋىرىڭ،مەن پاراڭ سالىۋىرەي،ئاڭلىغاچ قىلسىڭىزمۇ بولىدىغاندۇ؟
  _ھە،بولدى پاراڭلىشايلى،ئانچىمۇ مۇھىم ئەمەس،كېيىن قىلساممۇ بولىۋىرىدۇ.
  _ئۇنداقتىچۇ قاراڭ ئۇكام،سىزنى ئۆزەمنىڭ ئۇكىسىدەكلا كۆرىمەن،شۇڭا دەۋىرەي ھە،تۈنۈگۈن ئىدارىدىكى قىزلار بىلەن شاڭخەي مەدىنىيەت شەھەرچىسىگە بېرىپ كەلدۇق قاراڭ،ھېيىت ئەمەسمۇ،ئايال كىشى بولغاندىن كېيىن ئۈستىمىزگە بىرەر قۇر كىيمىسەك قانداق بولىدۇ دەڭا؟
  _ۋاي-ۋۇي ئاسىيە ھەدە،ئۇنىڭسىزمۇ كىيىملىرىڭىز بەكلا كۆپقۇ سىزنىڭ_دىدىم ئۇنىڭ سېمىز تۇرقىغا ماسلاشتۇرۇپ كۈندە بىر ياڭزا كىيىپ كېلىدىغان كىيىملىرىنى كۆز ئالدىمغا كەلتۇرۇپ.
  _ۋاي ئاللا،ئۇ دىگەن ھېيىتلىق ئالغان كىيىملەر بولمىسا،بۇ دىگەن ئۇلۇغ قۇربان ھېيىت تۇرسا،كىيمىسەك قانداق بولىدۇ؟
  _شۇنداقمۇ،ھېيىت-بايراملاردا چوقۇم يېڭى كىيىم كىيىش كېرەكمۇ ھەدە؟_مەن ھەيران قالغان ھالدا ئۇنىڭغا تىكىلدىم.
  _ئاللا،ئاللا نىگار،ماۋۇ جەمىيەت تەرەققى قىلىپ نەگە باردى؟سىزنىڭ كاللىڭىز نەدە؟_ئۇ مېنىڭ ئاددىغىنە پادىچىلار شىمى،ئۆرە ياقىلىق زەڭگەر گاڭسىرۋە پاشنىسى تۈز كەلگەن ئوقۇغىچىسىمان خۇرۇم ئايىغىمغا شۇنداق بىر نەزەر تاشلاپ،كىيىم ئاسقۇدىكى پەي چاپىنىمغىمۇ «يالىت»تىدە قاراپ ئۈلگۈردى،_تۈنۈگۈن قىزلارنىڭ سىزنى«مەتۇ»دىگىنى بىكار ئەمەسكەن جۇمۇ.
  -نىمە؟
  _ۋاي ئاللا،مېنىڭ مۇشۇ ئاغزىم،ھەي بۇ شۇ قاراڭ ئۇكام،_ئۇ ئالدىراپ ئېغىزىدىن چىقىرىۋەتكەن گەپكە سەل ھودۇققاندەك قىلىپ تەمتىرەپ كەتتى،_بۇ ئۇنداق دىگىنىم ئەمەس،شۇ ئەمدى،قىزلارنىڭ دىگىنى سىز دىگەن شۇنداق چىرايلىق،بوي-تۇرقىڭىزمۇ،پىگۇرىڭىزمۇ بەك قاملاشقان،سىزنىڭ ئۈستىڭىزگە چىققان ھەر قانداق ئېسىل كىيىممۇ ئۆز خىلىنى تاپقاندەك بولاتتىكەن،بىراق سىز شۇ بەك ئاددى يۈرىسىز،ئاددى-ساددىلىققۇ ياخشى خىسلەت ئۇكام،بىراق بىز بۇدۇنيادا قانچىلىك ياشايمىز دەيسىز؟توي قىلىپ بىر ئۆي بېشىڭىزغا كىيىلگەندە بىلىسىز تېخى،ئۇ چاغدىكى ئارمانلارچۇ،ۋاي يېتىشتۇرۇپ بولالماسلىقتىن جاق تويۇسىز،ئۇنىڭغا قىلسىڭىز بۇ تۇرغان،بۇنىڭغا قىلسىڭىز ئۇ تۇرغان ھەي..._ئۇ تۇرقىغا قىلچىمۇ ماس كەلمىگەن جاپاكەش ئاياللاردەك«ئۇھ» تارتىپ قويۇپ سۆزىنى داۋاملاشتۇردى-شۇڭىچۇ ئۇكام،مۇشۇ ۋاقىتلىرىڭىزنى قەدىرلىسىڭىز بولمامدۇ دەيمەن،بۇ دىگەن سىزنىڭ قورچاقتەك ياسىنىپ،نى-نى يىگىتلەرنىڭ كۆزىدىن ئوت چىقىرىپ تاقىلداپ يۈرىدىغان ۋاقتىڭىز،ئاتا-ئانىڭىز ئىككىلىسى چاقىرسا يەر تىترىگىدەك ئابرۇيلۇق مەنسەپدار ئادەملەر،ئۇلارنىڭ سىز نىمە دىسىڭىز ئاشۇرۇپ قىلىپ بەرگۈدەك بىر تال گۈلى تۇرسىڭىز،يەنە تېخى ئۆزىڭىزنىڭ خىزمىتى، مۇئاشى دىگەنلەر يېتىپ ئاشقۇچە تۇرۇپتۇ،ئاۋۇ لابىراتۇرىيە بۆلۈمىدىكى گۇلبەرنى كۆرگەنسىز،سىز بىلەن تەڭ دىمەت،سىزچىلىك پىگۇرى بولسىچۇ ئۆلەي،چېچىنى ھەپتىدە بىر ياڭزا بوياپ،كۈندە بىر ياڭزا كىيىپ ،ئىشى بولسۇن،بولمىسۇن بۆلۈملەرنى ئارىلاپ كىيگەن كىيىمى، قىلغان گىرىمى ،چاچ پاسۇنلىرىنى كۆز-كۆز قىلىپ يۈرۈدۇ.سىزنىچۇ ئۇنىڭغا ھەرگىز سېلىشتۇرغىلى بولمايدۇ ئۇكام،سىز شۇنداق يېقىشلىق، مۇلايىم،ئەممازە،مۇشۇ تۇرقىڭىزغا شۇ چاقناق كىيىملەر ماسلاشقان بولسا،ھەي-ھەي،_ئۇ تامىقىنى چىكىلدىتىپ تۇرۇپ سۆزىنى داۋام قىلدى_سىزنىڭ يىگىتىڭىزمۇ بەك كېلىشكەن ئىكەن ھە؟ ئاڭلىسام ئائىلىسى شۇنداق بىلىملىك،باي، ئەتىۋارلىق بىر تال ئوغۇل دەيدىغۇ؟؟
  مەن نىمە دىيىشىمنى بىلمەي ئاغزى-ئاغزىغا تەگمەي سۆزلەۋاتقان بۇ ئايالغا قاراپ خېجىللىق بىلەن باش لىڭشىتتىم.
_ھە شۇ،تۈنۈگۈن گۈلبەرنىڭ چىكىلداپ كەتكىنى بىكار ئەمەستە شۇ،ئۇچۇ قاراڭ.«ئېسىت،ئېسىت،شۇنداق كېلىشكەن بالا زايىلا كەتتى بۇ كىتاپ خالتىسىغا»دەپ بىر مۇنچە گەپ قىلدى،قارىسام ئۇنىڭ  ئۇ يىگىتتىن ئوبدانلا تاماسى باركەن،مەن ئويلىنىپ قالدىم قاراڭ،سىزنىڭ ئاشۇ سەت دومباقتىن قالغۇچىلىكىڭىز بارمىتتى؟سىزغۇ ھەممە تەرەپتىن ئۇنىڭدىن ئۈستۈن تۇرۇسىز،شۇ بىرلا يېرى ياسىنىپ يۈرمەيسىز،خۇددى كىچىك بالىدەكلا،سىزنى تونىمايدىغانلار كۆرسە ئىدارىدىكى بىرەرسىنىڭ ئوتتۇرا مەكتەپيىكى قىزىمىكىن دەپ قالىدۇ،ھا،ھا،ھا_ئۇ بۇ ئوخشىتىشىدىن ھوزۇرلانغان ھالدا كۈلۈپ كەتتى ۋە ئارقىدىنلا _كۆڭلىڭىزگە كېلىپ قالمىغاندۇ ھە؟_دەپ قوشۇپ قويدى.
  _ياقەي_مەن ئوڭايسىز ھالەتتە بېشىمنى چايقىدىم،دىمىسىمۇ مېنىڭ ئارامبەخىش (شىيۇ شەن)تۇرقۇمنى كۆرگەنلەرنىڭ تەبىئى ھالدا مېنى«ئوتتۇرا مەكتەپ ئوقۇغۇچىسى» دەپ قېلىشى تۇرغانلا گەپ ئىدى.
  _قاراڭ ئۇكام،گەپنىڭ پوسكاللىسىغا كەلسەك،بۇ يىل بۇلغارىيەنىڭ يېڭى پاسۇندىكى نوركا پەلتۇسى مودا ئىكەن.تۈنۈگۈن بىز شاڭخەي شەھەرچىسىدىن كۆرۈپ كەلدۇق،ۋاي،ۋاي،ۋاي،ئۇنىڭ ياقىسى،ئۆزىنىڭ يۇمشاق سىلىق يۇڭلىرى،ئېسىللىقىنى دىمەيسىز،ھەممىمىزنىڭ ئىچىگە ئوت-چوغ چۈشۈپ كەتتى قاراڭ،بىر نەچچىمىز ئالىدىغان بولۇپ سۆزلەشتۇق ،دۇكان ئىگىسى مېنىڭ كونا خېرىدارلىقىمنى كۆزدە تۇتۇپ ئەسلىدىكى ئون ئۈچ مىڭ سەككىز يۈز يۈەندىن سەككىزمىڭ بەش يۈز يۈەنگە بېرىدىغان بولدى دەڭا..
   _نىمە8500يۈەن؟_مەن چۆچۈپ كەتتىم،بۇ بىزنىڭ بەش ئايلىق مۇئاشىمىزغا باراۋەر ئىدى.
   _ھەئە،ئۇ دىگەن ئېتىۋار باھا جانىم،قايتىپ بېرىپ ئېرىمغا دىسەم،ئېرىم دىگەن لامزەللىگە نىمە بولغان دىمەمسىز ،ئۇنى دىسەم بۇنى،بۇنى دىسەم ئۇنى دەپ يەنە تېخى ئالدىنقى يىلى ئالغان نوركا پەلتۇلىرىمنى دەپ يۈرۈدۇ دەڭا،ئەتىگەندە يەنە سوقۇشۇپ چىقتىم،قىيىن ئاپام ئاغرىپ بەلنىستە ئەمەسمۇ،شۇنىڭغا پۇل ئىشلەتتىم،نەدىمۇ ئۇنداق كۆپ پۇل بولسۇن دەپ،ھەي بەزىدە مۇشۇ ئەر كىشى دىگەن ،ئايال كىشىنىڭ شۇ كىيىم-كىچەك،زىبۇ-زىننەتلەر بىلەن كۆڭلى ئۆسمىسە نىمە بىلەن كۆڭلى ئۆسىىدۇ دەيسىز_ئۇ بىردىنلا كۆزلىرىگە ياش ئالدى.ئۇنىڭ بۇنداق يولسىز كىيىم كىيىشىگە ئەسلىدىلا نارازى بولۇپ تۇرغان بولساممۇ يەنىلا باشقىلارنىڭ كۆز يېشىنى كۆرسەم چىدىمايدىغان تەبىئىتىم تۈپەيلى كۆڭلۈم ئېرىپ بىر خىل بولۇپ قالدى.«كىم بىلىدۇ،ئۇنىڭ تۇرمۇشىدا بىز بىلمەيدىغان قانداق ئىشلار بارلىقىنى؟تېشىدا كۈلۈپ يۈرگىنى بىلەن ئىچىدە ئۆزىلا بىلىدىغان سىرلىرى بولىشى مۇمكىنغۇ؟شۇ سىرلار تۈپەيلى ئۇ ئۆزىنى شۇنداق ياسىنىپ،كىيىنىپ،خوشال يۈرۈشكە ئۈندەمدۇ تېخى».
   _تۈنۈگۈن ئاياللارنىڭ ئالدىدا ئەتە ئالىمەن دەپ مەيدەمگە ئۇرۇپتىكەنمەن،ئەندى بۈگۈن ئالالمايمەن دىسەم نىمە سەتچىلىك؟ئۇنىڭ ئۈستىگە ئەتىگەنلا گۈلبەر«ھەدە،قاچان بارىمىز مەن پۇلنى تەيارلاپ قويدۇم»دەپ تېلىفۇن قىپتۇ،شۇنچە كىيىم كىيىپ مۇئاشىنى ئۇ ئايدىن بۇ ئايغا يەتكۈزەلمەيدۇيۇ نەدىن تېپىپ كەلدىكىن دەيمەن.ھەي شۇڭىزە ئۇكام،ئويلاپ ئاخىرى قېشىڭىزغا كىردىم،سىزگە ئىشىنىپ جۇمۇ،سىز بۇنى ھەم ھىچكىمگە دىمەيسىز،ھەم چوقۇم ياردەم قىلالايسىز،ئۆزۈمنىڭمىچۇ چېيىمنىڭ پۇللىرىدىن سىقىپ يىغىپ قويغان 3500يۈەن پۇلۇم بار،ئەمدى شۇ 5000يۈەن كاملاپ،سىزدە چوقۇم بار ھە ئۇكام،بېرىپ تۇرسىڭىز مەن ئاستا-ئاستا قايتۇرۇۋەتسەم،قانداق دەيسىز؟_بىر سائەتكە يېقىن داۋاملاشقان بۇ ئاغزى بېسىقماي سۆزلەشتىكى تۈپ مەقسەت ئاخىرى تېشىلدى.
  _ھەدە، يولدىشىڭىز بىلەن شۇ توغرىدا جىدەللىشىپ كەپسىز،ئەمدى ئۇنى ئېلىپ بارسىڭىز نىمە دەر؟پۇلنىغۇ بېرىپ تۇرۇشتىن قورىققىنىم ئەمەس،شۇ تۈپەيلى ئىككىڭلار ئازارلىشىپ قالساڭلار..._مەن تەڭقىسلىق ئىچىدە ئۇنىڭغا قارىدىم.
  _ئوھۇش ئۇكام،سىز بىلمەيسىز،ئەر كىشى دىگەننىڭچۇ ئاشۇنداق ئالىمەن دىگەننى ئېلىپ،كىيىمەن دىگەننى كىيىپ ئۆي تۇتقاندا ئاندىن گۈل كۆرۈنىسىز،بولمىسا كۈل،ئەر كىشى دىگەن خەقچۇ تومۇرچى كېلىدۇ،ماقۇل،ماقۇل،خوش،خوش دەپ تۇرسىڭىز بىر ئۆمۈر  ئۇچىڭىزغا يىپ چاغلىق نەرسىنىڭ چىقمىغىنى بىلەن كارى يوق،يەنە تېخى شۇنداق ياسىنىپ تاقىلداپ يۈرگەن خېنىملارنىڭ كەينىدىن چېپىپ سىزنىڭ ئۇنىڭ ئۈچۈن قورۇلۇپ،ماكچىيىپ قالغانلىقىڭىزنى نەزەرگە ئېلىپمۇ قويمايدۇ .ئېرىم بارغۇ ئۇنداق دىگىنى بىلەن ئارقامدىن كىيگەن كىيىملىرىمنىڭ ياراشقانلىغىدىن ھەۋەس قىلىپ يۈرۈدۇ.بۇندىن بۇرۇنمۇ نەچچە قېتىم شۇنداق بولغان،يانا تېخى مەغرۇر ئەر-ئاياللار سورۇنىغا مېنى قولتۇقلاپ كىرىشكە ئامراق،نىمىشكە؟مەن شۇنداق ياسىنىپ ئۆزۈمنى تاشلىماي يۈرگەچكە...ھى ھى ھى -ئۇ ھوزۇرلانغان ھالدا كۈلۈپ كەتتى.مەن  پۇلنى ئېلىپ بەرمەي بۇ پاراڭنىڭ ئايىغى چىقمايدىغانلىغىغا كۆزۈم يېتىپ تارىتمامدىن  مۇئاش چېكىمنى ئالدىمدە،ئىدارىمىزغا يېقىن يەردىكى بانكىغا قاراپ ماڭدىم.
    قولۇم دىگەندەك ئىشقا بارمىدى،ئارىلىقتا باشقىلارنىڭ قىيا_چۇۋالىرىغا قېتىلىپ ئاسىيەم ھەدە ۋە گۈلبەرنىڭ داغدۇغا بىلەن كىيىپ كەلگەن پەلتۇسىنى مۇبارەكلەپمۇ چىقتىم.ئەتىگەن قېلىپ قالغان ئىشىمنى تۈگىتىپ باشقا ئىشمۇ قىلمىدىم،قارىسام قىز بالا بولغاچقىمۇ ياكى ئەتىگەنكى ئاسىيەم ھەدىنىڭ گەپلىرى خېلىلا تەسىر قىلىپ قالدىمۇ مېنىڭمۇ تۇرۇپلا بازار ئايلىنىپ كىيىم كۆرۈپ باققىم كېلىپ قالدى.ئىشتىن سەل بالدۇر قايىتتىم،ئۇدۇل ئۆيگە كەتمەي ئايلىنىپ دۇكانلارغاكىرگەچ ماڭدىم.مۇۋاپىق بىرەر كىيىم ئۇچراپ قالسا ئېلىش ئويۇممۇ يوق ئەمەس ئىدى.شۇ خىيال بىلەن مېڭىپ قىزىل بايراق سودا سارىيىنىڭ يېنىغا جايلاشقان ئېسىل كىيىملەر مەركىزى بولۇپ قالغان قاتار دۇكانلار تەرەپكە قاراپ قەدەم تاشلىدىم.
   _ئاچا،ئاچا، ئازراق ياردەم قىلسىڭىز،قورسىقىم بەك ئېچىپ كەتتى، ئازراق تاماققا چۇشلۇق پۇل بەرسىڭىزچۇ،ئاچا_تۇيۇقسىز ئالدىمنى توراپ قوللىرىنى سۇنۇپ ماڭا ئۇمۇت بىلەن بىچارىلەرچە تىكىلىپ تۇرغان،چاچلىرى تارغاق سېلىنمىغىنىغا ئۇزۇن ۋاقىتلار بولغان،ئۇچىسىغا يىرتىق ،ئەسكى بىر نىپىز چاپان كىيىپ،سۇغۇق شامالدا تىترەپ ئەسلىدىلا قاراراق رەڭگىنى كۆكەرتىپ تېخىمۇ قارامتۇل بولۇپ قالغان قىزچاقنى كۆرۈپ ئىختىيارسىز تىترەپ كەتتىم.كۆزلىرىم ئۇنىڭ ئۇچى يىرتىلىپ كەتكەن چۇقاي كىيىۋالغان،يىرتىقىدىن بارماقلىرى كۆرۈنۈپ تۇرغان پۇتلىرىغا چۈشتى،ۋۇجۇدۇم قانداقتۇر بىر خىل سىلكىنىشتىن تىترەپ كەتتى.

  يول بويى قىزچاق كۆز ئالدىمدىن كەتمىدى،ئىچىمدە ئۆزۈمنى ئەيىپلەيىتتىم،«مەن قانداق ئادەم بولۇپ قالدىم؟بىچارە قىزچاق ،ئۇ قىزچاق مېنىڭ شۇ ئازغىنە پۇلۇمغىلا مۇھتاجمىتتى؟ مەن شۇ ئازغىنە پۇل بىلەن ئۇنىڭ يىتملىكنىڭ مۇشەققەتلىرى ئەزگەن قەلبىنى قانچىلىك خۇش قىلالىغاندىمەن؟ مەن ھىچ بولمىغاندا ئۇنى ئەگەشتۈرۈپ بىرەر ۋاق ئىسسىق تاماق ئېلىپ بېرىپ دۇكانغا باشلاپ ئەكىرىپ ئۇچىسىغا بىرەر قۇر ئىسسىققىنا كىيىم كىيگۈزۈپ قويالمىدىم ھە،مەن ئۇنىڭ ئاشۇ بىچارە،پاسكىنا يالىڭاچ تۇرقىدىن نۇمۇس قىلغاندىمەن،بولمىسا ئاۋۇ بىكار تەلەت بايۋەتچىلەرنىڭ كۇسۇرلاشقان مەسخىرە گەپلىرىگە نىمىشكە ئىككى ئېغىز بولسىمۇ رەددىيە بېرەلمىدىم؟قېنى مەندىكى ئادىمىيلىك،قېنى مەندىكى ھەققانى تۇيغۇلار،قېنى مەندىكى ئىسلام دىنىمىزدا تەكىتلىنىدىغان ھەقىقى مۇسۇلمان بەندىلىرىدە بولۇشقا تېگىشلىك پەزىلەت؟»
  مەن ئۆرتەنگەن ھالدا ئۆيگە كەلدىم، ئىشىكنى ئېچىپ كىرىپلا نازۇك تەبىئىتىمنىڭ غەلۋىسى بىلەن يىغلىۋەتتىم.ئۆيدە  نىمىنىدۇر قىزغىنلىق بىلەن پاراڭلىشىپ ئولتۇرغان ئاپام بىلان دادام چۆچۈپ ئورۇنلىرىدىن تۇرۇشۇپ ئالدىمغا كەلدى.
  _نىمە بولدى قىزىم؟
  _ئاپا، دادا مەن مەن ..._مەن گېپىمنى قىلالمايلا قالدىم....
بىر ئازدىن كېيىن ئۆزۈمنى تۇتۇۋىلىپ بۇگۈنكى ئىشلارنى بىر -بىرلەپ سۆزلەپ بەردىم،دادام كاللامدا بولۇۋاتقان«8500يۈەنلىك نوركا پەلتۇ»«كىيىم-كېچەكلىرى جۇل_جۇل قىزچاق»ۋەقەسىنىڭ توقۇنۇشىنىڭ مېنىڭ كەيپىياتىمغا نەقەدەر تەسىر كۆرسەتكەنلىكىنى ھىس قىلدى بولغاي ناھايىتى ئۇزۇن گەپ قىلدى،بۇ گەپلەر ئۆتمۈشتىن باشلىنىپ ھازىرقى بىزنىڭ مۇشۇ راھەت پاراغەتلىك تۇرمۇشىمىزغا كېلىپ توختىدى:
  _قىزىم،سىز گەرچە ئۇ قىزغا شۇ ئازغىنە پۇلنى تۇتقۇزۇپ قويۇپ ئۇنىڭغا باشقىچە ياردەم قىلالمىدىم دەپ ئۆكۈنۈپ شۇ سەۋەپتىن يىغلىدىم دىسىڭىزمۇ مەن قەلبىڭىزنى تامام چۈشەندىم، دەرۋەقە مۇشۇنداق تەڭسىزلىك بۇ جەمىيەتنىڭ ھەممە يەرلىرىدە مەۋجۇت،بۇنىڭغا گەرچە ئامال بولمىسىمۇ بىز كۈچىمىزنىڭ يېتىشىچە تىرىشساق بولىدۇ،سىز ئېيىتقاندەك ئاشۇ 12 ياشقا كىرگەن قىزچاق ۋە8 ياشلىق ئۇغۇلنىڭ ھازىر مەكتەپ قوينىدا بىلىم ئالىدىغان مەزگىلى،دادىسىنىڭ يوقلىقى،ئانىنىڭ كىسەل بولۇپ يېتىپ قېلىشى ئۇلارنى ئەنە شۇنداق مۇھتاجلىق قوينىغا سۆرەپ كىرگەن. دەرۋەقە بۇلارنىڭ تۇرمۇشى بىلەن ئاۋۇ تۇمەن يۇەنلىك كىيىم كىيىپ يۈرۈدىغان بايۋەتچىلەرنىڭ تۇرمۇشىدا ئاسمان-زىمىن پەرىق بار، بىز ھەممىز بىر نىيەت بىر مەقسەتتە ئورتاق تىرىشىپ ئۆزىمىزدىكى ھەقىقى ئادىمى خىسلىتىمىزنى ،پەزىلىتىمىزنى جارى قىلدۇرۇپ،يىتىم-يېسىر ئاجىزلارنى يۆلەپ،ئازغىنە مەنپەئەتىمىزنى قۇربان قىلالىغان بولساققۇ بۇنداق ئىشلار بۇنداق كۈنلەرمۇ بارا-بارا ئازىيىپ ئاخىرى يوقالغان بولاتتى.خەيىر ھەممىگە ئامال قىلالمىساقمۇ مۇشۇ ھېيىت مۇناسىۋىتى بىلەن ئاپىڭىز ئىككىمىز ئون مىڭ يۈەن پۇل چىقىرىپ ئۆزىمىز داۋاملىق ھالىدىن خەۋەر ئېلىپ كېلىۋاتقان بىر نەچچە ئائىلىلەردىن ھېيىتلىق ھال سوراپ كەلمەكچى ئىدۇق ،ئۇ قىزلارنىڭ ئۆيى قەيەردىكەن،خەيىر قىزىلبايراقنىڭ ئالدىغا بارساق تاپارمىز،ئۇلارنىمۇ بۇ قېتىمقى سانغا كىرگۈزەيلى.
   _دادا،ئاپا، مەن ئەتىگەن كىيىم ئالايمىكىن دەپ چىقارغان ماۋۇ ئىككى مىڭ يۈەننىمۇ قوشاي!مەن ھېيىتلىق كىيىم كىيمەيمەن.
  -سىزنىڭ مۇشۇنداق ئاددى -ساددا تۇرقىڭىز ماڭا گۈل كۆرۈنىدۇ قىزىم،مۇھىمى سىزدە مەن ئۈمۈت قىلىدىغان ئېسىل خىسلەتلەر بار، مەن شۇنىڭ ئۈچۈن سىزدىن پەخىرلىنىمەن، ھەر قانداق ئورۇن ھەر قانداق جايدا سىزدەك ئاددى «ئوتتۇرا مەكتەپ ئوقۇغۇچىسى،يەنە تېخى مەتۇ قىزىم»بولغانلىغىڭىزنى كۆرسىتىپ قويغۇم كېلىدۇ تېخى،_دادام ھاياجان بىلەن ماڭا قاراپ خۇددى كىچىك ۋاقىتلىرىمدەك قۇچىقىغا ئولتۇرغۇزۇۋىلىپ بوينۇمنى غىدىقلاشقا باشلىدى.«ۋاي دادا، قىچىقىم كېلىپ كەتتى »مەن قاقاقلاپ كۈلگىنىمچەتېپىرلاشقا باشلىدىم،ئاپام ماڭا ياردەم بەرمەكچى بولۇپ كېلىپ دادامنى قىچىقلىدى...ھا.ھا.ھا  ،خوشاللىق كۈلكە سادالىرىمىز تاملارنى يېرىپ ئۆتۈپ زاللارغا تارقالدى..كۆزلىرىمگە نۇرغۇنلىغان يېڭى كىيىم كىيپ خوشاللىق ناخشىلىرىنى ياڭرىتىۋاتقان سەبى بالىلار مەكتەپ مەيدانىدا بىزنى ئوراپ ئۇسۇل ئويناۋاتقاندەك كۆرۈنۈۋاتاتتى....
