{ ئايچىچەك } قىزلار بۇ دۇنيانىڭ جۇلاسى

بۇ يازما nawa717 تەرپىدىن نادىردىن قالدۇرۇلدى(2008-03-24)

خۇددى ئاسماننىڭ جۇلاسى قارا پەردىگە ئالماس قاداپ قويغاندەك چاراقلاپ يىنىپ تۇرغان سانسىز يۇلتۇزلاردەك بولغىنىدەك،چىن مەنىسىدىن ئىيىتقاندا بۇ دۇنيانىڭ،ھاياتلىقنىڭ ھەقىقى جۇلاسى گۈل يۈزلۈك ئاشۇ قىزلاردۇر.
          دىنىي رىۋايەتلەردگە قارىغاندا،ئادەم ئاتا يارىتىلىپ ئۇزۇن ئۆتمەي ئۇنڭ تەبئىيتى ۋە سسغىنىشىنىڭ تەقەززاسى تۈپەيلىدىن ئۇنىڭ يىتىم قوۋۇرغىسىدىن ئىنسانلارنىڭ تۇنجى ھەۋۋا ئانا يارىتىلغان.ئۇلار مەيلى جەننەتتىكى بەخىتلىك كۈنلىرىدە بولسۇن ۋە مەيلى گۇناھ ئۆتكۈزۈپ زىمىنغا ھەيدەلگەندىن كىيىنكى جاپالىق كۈنلىرىدە بولسۇن،ھايات-ماماتتا بىرگە بولۇپ،پەرزەنىت قالدۇرۇپ،ئىنسان ئەۋلادىنڭ بۈگۈنگۇچە داۋاملىشىپ كېلىشىنى كاپالەتكە ئىگە قىلغان.خۇددى ئادەم ئاتا تەشنا ۋە تەقەززار بولغىنىدەك،قىزلار بۇ دۇنيانىڭ تەقەززاسى ۋە تەشنالىقى،تەشنالىقنى قاندۇرغۇچى،شۇ تەشنالىق ۋە قىزغىن مىھرى-مۇھەببەتقاينىمىدا ئەۋلاد قالدۇرغۇچى ھەرنىمە دەپ تەرىپلىسىمۇ ئەرزىيدىغان پەزىلەت ئىگىسىدۇر.شۇنداقلا گۈزەللىكى،نازۇكلۇقى،جەلىپكارلىقى،نازاكەتلىكلىكى ،پەزىلىتى بىلەن بىردەستە ئەجايىپ خۇشپۇراق گۈلگە،جىرىڭلاپ تۇرىدىغان جانان چىنىگە،سۈزۈك ۋە تىنىق قاشتىشىغا ئوخشايدۇكى ئۇنى ئاۋايلاپ پەرۋىش قىلىش،ئاسراشقا خەلقىمىز ئەزەلدىن كۆڭغل قويغان،مول تەجىربە ھاسىل قىلغان،بۇ جەھەتتە ئۆزىگە خاس مەدىنىيەت ئەندىزىسى ياراتقاندۇر. ھەربىر ئاتا-ئانا قىزلىرىنى ئەدەپلىك،ئەخلاقلىق،مىھرىبان،ئۇز خۇيلۇق قىلىپ تەربىيەلەشكە ئەھمىيەت بىرىپ كەلگەندۇر.  
          خەلقىمىزنىڭ ئۇزاق ئۆتمۈشىدىكى ئەپسانە-رىۋايەتلىرىدە ئۇيغۇر قىزلىرى نۇردىن،خىسلەتلىك دەرەخلەردىن تۆرىلىدۇ ۋە پەزىلەت غۇنچىسىنى قەلىپ تۆرىدە ئىچىلدۇرۇپ،ناھايىتى پاك،مەسۇم ۋە ئادەمنىڭ ئىچى كۆيگۈدەك بولۇپ ئۆسۈپ يىتىلىدۇ. ئوغۇزخاننىڭ نۇر ئىچىدە تۆرەلگەن خوتۇنى پاكلىق ۋە گۈزەللىكتە ناھايىتى يۇقىرى سۈپەتلەرگە ئىگەبولۇپ،«ئۇ كۈلسە كۆكمۇ كۈلىدۇ ،ئۇ يىغلىسا كۆكمۇ قوشۇپ يىغلايدۇ». ئۇنىڭدىن كىيىن ئۆتكەن تۇمارىسقا ئوخشاشمىڭلىغان باتۇر،نازاكەنلىك،گۈزەل ئۇيغۇر قىزلىرى تارىخ بىتىدىن مۇناسىپ ئورۇن ئىلىپ،ئۇيغۇر قىزلىرىنىڭ شەنىنى كۆتۈرگەن،ئابرويىنى كۆككىچە كۆتۈرگەن. ئورخۇن بويلىرىدا ئات چاپتۇرۇپ سەلتەنەت سۈرۈپ ئۆتكەن ئەجداتلىرىمىز دەۋرىدە قاغانلار پەرمان ئىلان قىلغاندا «قاغان پەرمان قىلىدۇكى....»دەپ ئىلان قىلسا كۈچكە ئىگە بولمايىتتى،پەقەت «قاغان بىلەن خانىش پەرمان قىلىدۇكى،،،»دەپ ئىلان قىلغاندىلا كۈچكە ئىگە بولاتتى.شۇ دەۋىرلەردە تاكى ئىسلام دىنىغا كىرگۈچىلىك ئۇيغۇر ئەجداتلىرىنىڭ مەيلى باي مەيلى كەمبەغەل ۋە مەيلى ئەمەلدار بولىشىدىن قەتئىنەزەر،بىر ئەر ،بىر خوتۇنلۇق بولۇشتا ئىزچىل  چىڭ تۇرىشىمۇ خەلقىمىزنىڭ قىزلارنى ئالاھىدە ھۆرمەتلەيدىغانلىقى ۋە .ئىززەتلەيدىغانلىقىدىن دىرەك بىرىپ تۇرۇپتۇ.  ھالبۇكى،جاھالەت پىرلىرى بولمىش خوجا -ئىشانلارنىڭ ئۇزۇن مۇددەت جاھالەتتە قالدۇرىشى تۈپەيلى،قىزلىرىمىزنىڭ قەدىر-قىممىتى ئاجايىپ زور دەرىجىدە تۆۋەنلەپ كەتتى،ئاشكارا ھالەتتىن بىكىندۈرۈلدى،ھەتتا ساۋاتلىق بولۇش ھوقوقىدىنمۇ مەھرۇم قىلىندى،ئەمما خەلقىمىزنىڭ قېنىغا تىندۇرما بولغان قىزلارنى ئەتىۋارلاش ۋە ئۇلارنى كۆيۈنۈپ تەربىيەلەشتىن ئېبارەت ئېسىل خىسلەت زامانلارنىڭ ئۆتۈشى بىلەن سۇسلىشىپ قالمىدى،باشقا مەدىنىيەت تىپىدىكى خەلىقلەرگە ئوخشاش قىز پەرزەنىت كۆرسە ئۇنى پەس كۆرىدىغان،خورلايدىغان،ھاقارەتلەيدىغان،يۈك دەپ قارايدىغان ئىجدىمائىي كەيپىيات ئۇيغۇر جەمىئىيتىدە چۇڭقۇر يىلتىز تارتالمىدى.
          ھالبۇكى،فېئوداللىق ئىددىينىڭ تەسىرىدە بەزى قىزلىرىمىز ئۆزلىرىنى تۆۋەن كۆرىدىغان،ئۆزىنى ئۆزى كەمسىتىدىغان،چوڭ ئىشلارنى قىلالمايمەن،قىزلارنىڭ قولىدىن ئىش كەلمەيدۇ ،دۇنيانىڭ مەزلۇمىبىز،نېمىشقا ئوغۇل بولۇپ يارىلىپ قالمىغاندىمىز ،دەيدىغان تولىمۇ خاتا ۋە بىر تەرەپلىمە قاراش تۈپەيلىدىن ھاياتلىق يولىدا ئۆزىنى ئۆزى چۈشەپ قويىدىغان ،ئۆقەدىر-قىممىتىنى ئاشۇرۇش ئۈچۈن يول تاپمايدىغان ،ئىزدەنمەيدىغان،ئىنتىلمەيدىغان خاھىش ناھايىتى كۈچلۈك بولۇپ كەلگەنىدى.  20-ئەسىرنىڭ ئالدىنقى يىرىمىدا شىنجاڭنىڭ ھەرقايسى جايلىرىدا يېقىلغان مەرىپەت چىراغلىرى مەھكۇملۇقتا ئىچى سىقىلغان قىزلىرىمىزنىڭ قاراڭغۇ دىللىرىنىمۇ يورۇتۇپ،ئۇلارنىڭ زامانىۋى دۇنيا بىلەن ئۇچرىشىشنى ،بىلىم ئېلىشنى،مۇشۇ ئارقىلىق ئەۋلادلارنى ياراملىق قىلىپ تەربىيلىشىنى كاپالەتكە ئىگە قىلىپ،خەلقىمىزنىڭ مەنىۋى ساپاسىنىڭ ئوموميۇزلۈك يۇقىرى كۆتۈرلىشىگە تۈتكە بولغانىدى. بۇنىڭ نەتىجىسىدە قىزلىرىمىزنىڭ يۈزىنى يىپىپ تۇرغان قارا چۈمپەردە ئىلىۋىتىلدى،قىزلار ئىنتىلگىنىچە،ئارزۇ قىلغىنىغا قەدەر يۇقىرى ساپاغا ئىرىشىش ئىمكانىيىتىگە ئىرىشتى. مۇشۇنداق ئەۋزەل شارائىتتا،قىزلارنى ئەزىزى كۆرگەن خەلقىمىز قىزلىرى ئۆزلىرىنىڭ قىز بولۇپ يارىلىپ قالغىنىدىن ھەرگىز ئۆكۈنمەسلىكى،ئەكسىچە پەخىرلىنىشى،مەغرۇرلىنىشى كىرەك. چۈنكى،قىزلار بۇ دۇنيانىڭ جۇلاسى ھەم بىزىكى،ھاياتلىق قۇۋانچىسى،ئىنسان ئەۋلادىنىڭ داۋاملاشتۇرغۇچىسى،شۇنداقلا ئوغۇللاردىن ئۈستۈن تۇرىدىغان سېھرىي قۇۋەتكە ئىگە ئىنساندۇر. 
        بىر نەچچە يىل ئىلگىرى مەن بىر ژورنالدىن مەملىكىتىمىزدىكى ئون مەشھۇر كارخانچى قاتارىدىن ئورۇن ئالغان بىر ئايال كارخانچىنىڭ مۇۋەپپەقيەت قازىنىش يولى ئۈستىدە توختىلىپ:«مەن بىر ئانا، مەن بالا تۇغالىغان ئىكەنمەن، ئەرلەر قىلالىغان ھەرقانداق ئىشنى قىلالايمەن»دېگەن ئالتۇندەك سۆزنى ئوقىغانىدىم. قىزلار تەڭرى تەرىپىدىن مۇشۇنداق تەڭداشسىز خىسلەتكە ئىگە قىلىپ يارىتىلغانىكەن، ھەرگىز ئۆز-ئۆزىنى كەمسىتىش پوزىتسىيسىدە بولماي، ئۇنىڭ ئەكسىچە رېئاللىققا يۈزلىنىپ،ئەمدى نىمە قىلىمەن،كەلگۈسىدە قانداق قىلىمەن؟دېگەننى كۆپرەك ئويلىشى ،ئۆزىنى چۈشىنىشنى ئىلگىرى سۈرۈپ،بىلىمنىڭ يۈكسەك چوققىسىغا  ئىنتىلىپ، قىزلار ھازىرلاشقا تىگىشلىك شەرتلەرنى پۈتۈنلەي ئۆز ۋۇجۇدىدا ھازىرلاپ،ئۆزىنى تاكامۇللاشتۇرۇپ،كامىل ئىنسان بولۇپ يىتىلىشكە بەل باغلىشى كىرەك.  
         ھەي،گۈزەل قىز ! سىز شۇ تاپتا مەلۇم بىر ئۇنۋېرسىتنىڭ ئازادە كوچىسىدا زامانغا لايىق كىيىنىپ مەغرۇر قەدەم ئىلىپ كىتىۋىتىپ،بۇنىڭدىن كىيىنكى ھاياتلىق يولىڭىزدا قانداق قىلىشنى،قانداق ئىستىقبالغا ئىرىشىشنى كۆزلىگەن،كۆڭلىڭىزگە نۇرغۇن يۈكسەك غايىلەرنى پۈككەن،شۇ يولدا تىنىمسىز تىرىشىۋاتقان،ئىزدىنىۋاتقان،ئويلىنىۋاتقان غايىلىك ئوقۇغۇچى بولىشىڭىز مۇمكىن ،ياكى ئوتتۇرا مەكتەپتەئوقۇۋاتقان،ئالىي مەكتەپلەردە ئوقۇشنى ئارزۇ قىلىۋاتقان ئىنتىلىشى كۈچلۈك،ئۇلۇغ غايىگە ئىگە ئوقوغۇچى بولىشىڭىز مۇمكىن ۋە ياكى ئۆرلەپ ئوقۇش ئارزۇيىڭىز ئەمەلگە ئاشماي قالغان ، ئاساسىي قاتلاملاردا ئىشلەۋاتقان ياكى خىزمەت كۈتۈپ تۇرغان ۋە ياكى ھاياتلىق يولىدا تەمتىرەپ قالغان قىز بولىشىڭىز ھەم مۇمكىن. ھەتتا نۇرغۇن ئارزۇ-ئارمانلىرىڭىز مۇرەككەپ ئىجدىمائىي سەۋەپلەر تۈپەيلىدىن كۆپۈككە ئايلانغان، ئارمىنىڭىز ئارمان پىتى قالغان،تەكلىماكاننىڭ بىر چىتىدىكى خىلۋەت ماكانغا باغلىنىپ قالغان، ئۈنۈمسىز يەرلىرىڭىزگە تەر تۆكۈۋاتقان دېھقان قىزى بولىشىڭىز مۇمكىن.  قانداقلا بولىشىدىن قەتئىنەزەر ،سىز گۈزەل،نازاكەتلىك،ياخشى خۇلۇقلۇق بىر ئۇيغۇر قىزى بولغانىكەنسىز،ئۆز مىللىتىمىزنىڭ ئەنئەنىسىى بويىچە ئۆز-ئۆزىڭىزنى تاكامۇللاشتۇرىشىڭىز كىرەك،ھەرگىز بۇ قۇتلۇق يولدىن قايماسلىقىڭىز كىرەك.  ئۆزلۈكسىز ئىلىم ئۈگىنىشكە ئادەتلىنىپ،كەلگۈسىدە ئۆزىڭىزگە ئاجايىپ يۇقىرى ساپا،جەمىئەتكە مىننەتسىز تۆھپە قوشالايدىغان روھ، كەلگۈسىدە ئەۋلادلىرىڭىزنى ياراملىق قىلىپ تەربىيلەشكە باب كىلىدىغان مەنىۋى قۇرۇلما ھازىرلىشىڭىز كىرەك.  
        قىزلار دۇنياسىدا قانداق ئۆزگىرىشلەر بارلىققا كىلىۋاتىدۇ،سىز بۇ ئۆزگىرىشلەرگە قانداق قارايسىز،قانچىلىك ماسلىشىپ كىتەلىدىڭىز؟ بۇ ئۆزگىرىشلەرنىڭ ئىجانىي تەرەپلىرى قانچىلىك،ئەگەر بۇ ئۆزگىرىش مۇشۇ پېتى كىتىۋەرسە قانداق يامان ئاقىۋەتلەرنى ئىلىپ كىلىشى مۇمكىن؟ قانداق مەنىۋى كەيپىياتبۇنداق يامان ئاقىۋەتلەرنى ئۈنۈملۈك توسۇپ قالالايدۇ؟  ئەي قىز، مانا بۇ سۇئاللار ئالدىڭىزغا قويۇلۇپ باقتىمۇ، بۇ مەسىلىلەرنىڭ بىرەرى توغرىسىدا باش قاتۇردىڭىزمۇ،قانداق خۇلاسىلەرگە كالدىڭىز؟ 
        دەرۋەقە ئالدىڭىزغا قويۇلغان سۇئاللار ئىنتايىن نۇرغۇن ،ئۇنىڭ جاۋاپلىرىمۇ ناغايىتى نۇرغۇن بولىشى مۇمكىن. ھايات مۇمكىنچىلىكلەر ئىچىدىن يول تىپىپ ئۆزىنىڭ قانۇنىيىتى بويىچە ئىلگىرلەپ كېتىۋىرىدۇ. ۋاقىتقا يۈگەن سىلىشنى، ئۆز قىممىتىڭىزنى تېپىشنى بىلەمسىز ۋە شۇنداقلا قىزلارغا خاس گۈزەل ئەخلاقنى ئۆزىڭىزگە يار قىلىپ ئالماستەك ئىېرادە تىكلىيەلىسىڭىز،سىزنىڭ قىزلىق لاتاپىتىڭىز ۋە نازاكىتىڭىزدىن جاھان تەسىزلىنىپ پۈتكۈل  
موھىتىڭىز باھار ئەييامىغا جور بولىدۇ، بەخىتنىڭ نەغمە ناۋاسىنى ياڭرىتىپ،ھايات سازىنى سازلىيالايسىز!



قىز بولۇپ يارىلىش پىشانىڭىزگە پۈتۈلگەنىكەن، سىز بۇ مۇقەددەس ئىرادىدىن ئۆزىڭىزنى چەكسىز بەخىتلىك سانىشىڭىز، قىز بولغانلىقىڭىزدىن پەخىرلىنىشىڭىز،ۋۇجۇدىڭىزغا قىزلىق پەزىلەتنى جەملەپ،سېھرىي كۈچىڭىزنى قېزىشىڭىز،ھاياتتا ئىنسانلارغا پايدىلىقلا بولىدىكەن،ھەرقانداق ئىشقا ئىجتىھات قىلىپ،ئۆزىڭىزنىڭ ئەرلەرمۇ قىلالمايدىغان ئىشلارنى قىلالايدىغانلىقىڭىزنى بىلدۈرۈپ قويىشىڭىز كىرەك، ئۆزىڭىزنىڭ ھەيران قالارلىق نەتىجىلىرى ئارقىلىق كىشىلەرنى قايىل ۋە مايىل قىلالىسىڭىز، ئەي بەخىتلىك قىز، سىز بۇ دۇنيادا مەنىلىك ياشىغاننىڭ ھەقىقى پەيزىنى سۈرگەن ھىساپلىنىسىز.         ئۇچۇر دەۋرىنىڭ خىيالدىنمۇ تىز قەدەملىرى ئىشىك ئالدىغىلا كىلىپ قالدى، ئۇ سىز ئۆز قەدىمىگە ماسلىشىپ يولغا چىقىشقا تەقەززا قىلماقتا، ھالبۇكى سىز يەنىلا ئۈگىنىپ قالغان خۇي بىلەن سۆرەلمىلىك قىلامسىز ياكى ئۇنىڭغا ماسلىشىپ ، ئۇچۇر دەۋرىنىڭ ھەرقانداق يۇقىرى شەرتلىرىنىڭ ھۆددىسىدىن چىقىپ ئالغا ئىلگىرلەپ كىتەلەمسىز؟  مىللىتىمىزنىڭ ئەنئەنىۋى ئەخلاق-پەزىلەتلىرىنى تىخىمۇ ياخشى جارى قىلپ، ئۆزىڭىزنىڭ چولپاندەك پارلاپ تۇرغان قىزلىق نازاكىتى ۋە بۇ نازاكەت بەخىش ئەتكەن لاتاپەت بىلەن قىزلار دۇنياسىدا يۈز بىرىۋاتقان غەيرى ئۆزگىرىشلەرگە، ئەخلاق جەھەتتىن بۇزۇلۇشلارغا ئەمىلىيىتىڭىز ئارقىلىلىق جەڭ قىلالامسىز؟        سىز مانا شۇنداق قىلالايسىز، چۈنكى سىزنىڭ ۋۇجۇدىڭىزدا چەكسىز سىھرىي كۈچ كۆمۈلۈپ يىتىپتۇ، قىزلىق ئەدەپ ئەخلاق گۈللىرى خۇشپۇراق چىچىۋىتىپتۇ، ئازىراق تىرىشچانلىق كۆرسەتسىڭىزلا نازاكەت ۋە لاتاپەت بابىدا يىگانە چولپان بولۇپ قالىسىز، چۈنكى سىز دىگەن بۇ دۇنيا جۇلاسى!

    1-پەقەت بىر خۇتۇنلۇق تۇزۇملا قىز ئاياللارنى ئەزىزلەمدۇ؟
    2-ئۇيغۇر قىز -ئاياللىرى 20-ئەسىردىلا ئاندىن مەھكۇملۇق قاراڭغۇلۇقتىن قۇتۇلدىمۇ؟
    3-قىزلارنى تىرىك كۇمۇش-قىزلارنى پەس كۆرۇش ئەسلى كىملەرگە مەنسۇپ؟
    4-زىيادە چەكتىن ئاشۇرۇپ ماختاش،ھەتتائۇغۇللاردىن ئۇستۇن تۇرۇدۇ دەيدىغانغا نىمە ئاساسىڭىز بار؟باشتا ئادەم ئاتىنىڭ قۇۋرىغىسىدىن يارتىلغان ئايال،ئەجىبا ئاخىرىدا نىمىشقائۇغۇللاردىن ئۇستۇن بوپقالدى؟
