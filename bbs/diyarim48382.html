 شاڭخەي مۇسۇلمانلار مۇلازىمەت مەركىزى-مۇھەممەدئەلى ئەپەندىنى زىيارەت

0
شاڭخەي قىران سودا چەكلىك شىركىتىنىڭ باش دىرىكتۇرى «مۇسۇلمانلار مۇلازىمەت مەركىزى» قۇرغۇچىسى مۇھەممەدئەلى ئەپەندىنى زىيارەت قىلىش خاتىرسى
  
    ئەسسالامۇ-ئەلەيكۇم مۇھەممەدئەلى ئەپەندى، ئالدى بىلەن زىيارىتىمىزنى  قۇبۇل قىلغانلىقىڭىزغا چىن كۆڭلۈمدىن رەھمەت ئىيتىمەن .
   ئاساسى مەخسەتكە كەلسەك:
    خىزمەت مەسىلىسى نۆۋەتتە ئالىي مەكتەپنى پۈتتۈرىۋاتقان سىتۇدىنتلار ئۈچۈن بىر چوڭ مەسىلە بولۇپ قالدى،  سىزمۇ ئوقوغۇچىلىق تارىخىڭىزنى باشتىن كەچۈرۈپ،تىزلا شىركەت قۇرۇپ كۈرۈنەرلىك نەتىجىلەرنى قولغا كەلتۈرۈۋىتىپسىز، تىرىشچانلىقىڭىزغا بارىكاللا، ئەلۋەتتە مەنمۇ شىركىتىڭىزنىڭ تىخمۇ روناق تىپىپ دۆلەت ئىچى ۋە سىرتىغا يۈزلىنىشىنى مۇۋاپىقىتىنىڭ تىخىمۇ زور بولىشىنى ئۈمىد قىلىمىز. سىز بىر نەتىجە قازىنىش ئالدىدا تۇرغان ھەم بارلىق سىتودىنىتلىرىمىز ئۆلگىسى بولۇش سۈپىتىڭىز بىلەن ئۆزىڭىز، شىركىتىڭىز ۋە شىركەت قۇرۇش جەريانىڭىز ھەققىدە سۆزلەپ بەرسىڭىز. شۇنداقلا نۆۋەتتە ئىش كۈتۈپ تۇرىۋاتقان ياكى ئوقوش پۈتتۈرۈش ئالدىدا تۇرىۋاتقان سىتۇدىتنلارغا قانداق كۆڭۈل سۆزلىرىڭىز بار ئاڭلاپ باقساق دىگەن ئۈمىد تە.

مۇھەممەدئەلى ئەپەندى :ئەسسالامۇ ئەلەيكۇم،  مەن سىزنىڭ ماڭا تىلفۇن قىلىپ مەندىن مەن ۋە شىركەت ھەققىدە ئەھۋال سورىغانلىقىڭىزغا ھەم
شۇنداقلا مىنى زىيارەت قىلماقچى بولغانلىقىڭىزغا چىن  كۆڭلۈمدىن تەشەككۈرۈمنى بىلدۈرىمەن .
(زىيارەت چ چ دا يازما سۆھبەت ھالىتىدە ئىلىپ بىرىلغان)

سۇئال: مۇھەممەدئەلى ئەپەندى ئالدى بىلەن ئۆزىڭىزنى بىر تونۇشتۇرۇپ ئۆتكەن بولسىڭىز؟
جاۋاپ: ئىسمىم مۇھەممەدئەلى ئوسمان، 1980-يىلى 6-ئاينىڭ 26-كۈنى قەشقەر ۋىلايىتىنىڭ قاغىلىق ناھىيىسدە بىر دوختۇر ئائىلىسىدەتۇغۇلغان .بىز نىڭ ئائىلىدە 6 ئۇغۇل 1قىز پەرزەنت بولۇپ مەن بالىنىڭ ئالتىنجىسى ، بەش ئاكام بىر سىڭلىم بار. بۇ يىل 29 ياشقا كىرگەن بولساممۇ تىخىچە بويتاق (خىزمەت ھەلەكچىلىكىدە)
مەن باشلانغۇچ ۋە ئوتتۇرا مەكتەپلەرنى قاغىلىقتا تاماملىدىم، 1999-يىلى 9–ئايدا ئەلا نەتىجەبىلەن شىنجاڭ ئونۋىرسىتىتى ئىلىكتىرون قاكولتىتىنىڭ
" خەۋەرلىشىش قۇرۇلىشى" كەسپىگە قوبۇل قىلىندىم،،2004-يىلى شىنجاڭ ئونۋىرسىتىتىنى پۇتتۇردۇم ،

سۇئال: ھازىر سىتودىنىتلار ئونۋىرسىتىتنى پۇتتۇرگەندىن ئالدى بىلەن خىزمەت ئىزدەيدۇ،  سىزمۇ ئونۋېرستىتنى پۇتتۇرگەندىن كىيىن شىنجاڭدا خىزمەت ئىزدەپ باقتىڭىزمۇ ؟ نىمە ئۈچۈن شاڭخەيگە بىرىپ ئوقۇدىڭىز ؟
جاۋاپ: مەنمۇ ئەينى ۋاقىتتىكى ساۋاقداشلىرىم ۋە باشقا دوسلىرىمغا ئوخشاش خىزمەت تىپىش ئۈچۈن ئىزدىنىپ باقتىم،ھەم ھۆكۈمەت مەمۇرى ئورگىنىدىن خىزمەتمۇ تاپالىدىم ، لىكىن ئەينى ۋاقىىتا تاپقان خىزمىتىمنىڭ ئۆزەمگە ماس كەلمەيدىغانلىقىنى ھەم شۇنداقلا مەن بۇ خىزمەتنى داۋاملىق ئىشلىسەم ئۆزەمنىڭ غايىلىرىمنىڭ ئەمەلگە ئاشمايدىغانلىقىنى ھىس قىلىپ ،بۇ خىزمەتتىن ۋاز كەچتىم ، شۇنداقلا ئۆزەمنىڭ غايىسى ئۈچۈن چوڭقۇر ئويلاندىم ۋە ئىزدەندىم، مەن ئونۋىرىستىتنى پۇتتۇرۈپلا شاڭخەيگە چەتئەل تىلى ئۈەىنىش ئۈچۈن كەلدىم ، "شاڭخەي چەتئەل تىلى ئونۋىرسىتىتى" ۋە "توڭجى ئونۋىرسىتىتى "دا ئنگىلىس تىلى ۋە گىرمان تىلى ئۈگەندىم،

جاۋاپ : بىزنىڭ شىركىتمىز يىڭىدىن قۇرۇلغان بولغاچقا بىزدە مەبلەخ جەھەتتە ئازراق قىينچىلىق بار، ئەمما بۇ ۋاقىتلىق مەسىلە ، بۇنىڭ ھەل بولۇپ كىتىدىغانلىقىغا ئىشىنىمىز ، چۇنكى ھازىر بىر قانچە ئۇيغۇر سودىگەرلىمىز بىزگە مەبلەخ سىلىپ ھەمكارلىشىش توغرىسىدا بىز بىلەن سۆھبەت ئۆتكۈۋاتىدۇ ،بىز ھازىرئەمىلى خىزمەت ئىھتىياجىنىڭ ئىشىشىغا ئەگىشىپ تىخىمۇ كۆپ ئىختىساس ئىگىلىرىگە مۇھتاج. ئۇنىڭدىن سىرىت بىز شىنجاڭدىكى مۇسۇلمانچە يىمەكلىك ئىشلەپچىقىرىدىغان كارخانىلارنىڭ بىز بىلەن ھەمكالىشىشىنى ئۇمىت قىلىمىز ،بىز مۇ ئەڭ چوڭ تىرىشچانلىقىمىز بىلەن شىنجاڭدىكى مۇسۇلمانچە يىمەكلىك ئىشلەپچىقىرىدىغان كارخانىلارنىڭ مەھسۇلاتلىرىنى دۇنيا يۈزلەندۈرىمىز

سۇئال سىزنىڭ ھازىر يەنە قاندق پىلانلىرىڭىزبار؟
جاۋاپ: شىركىتىمىز نىڭ بىرقانچە ياخشى پىلانلىرى بار بولۇپ، بۇلارنىڭ ئىچىدە ئىش باشلاپ بولغىنىمىزدىن "تەرجىمان ئىجارىگە بىرىش مەركىزى" بولۇپ بىز شاڭخەي دىكى شىنجاڭلىق ئوقوغۇچىلاردىن بىر تەرجىمان ئىجارىگە بىرىش مەزكىزى قۇرۇش خىزمىتىنى باشلىدۇق، ھازىر شاڭخەيدە ئىنگىلىس تىلى ، ياپۇن تىلى ، فىرانسوز تىلى ،نىمىس تىلى قاتارلىق چوڭ تىللارنىڭ تەرجىمانلىرى تەمىنلەش  ئىھتىياجدىن ئىشىپ كەتكەن ھالەتتە، ئىش ھەققىمۇ باشقا كىچىك تىللارغا سىلىشتۇرغاندا كۆپ تۆۋەن، بىز شاڭخەيدىكى ھۆكىمەت ئورۇنلىرى، ساياھەت كوپىراتىپلىرى، كۆزگەزمە مەركەزلىرىنى ئەرەپ تىلى ، تۈرك تىلى ،روس تىلى ،قازاق تىلى ، ئۆزبەك تىلى ،قىزغىز تىلى ،تاجىك تىلى ،ئەزەربەيجان تىلى ،تۇكمەنىستان تىلى ،پەنجاپ تىلى قاتارلىق كىچىك دۆلەتلەتنىڭ تەرجىمانى بىلەن تەمىنلەيمىز ،ھازىر شاڭخەيدە ئىنگىلىس تىلى تەرجىماننىڭ بىر كۈنلۇك ئىش ھەققى 200 يۇەندىن ئاشمايدۇ، ئەمما ئاددىسى قاراق قىرغىز ئۆزبەك تىللىرى تەرجىمانىنىڭ كنلۇك ئىش ھەققى 400-500 يۇەن ھەتتا ئۇنىڭدىن كۆپرەك بولىدۇ، بىز دە بۇخىل ئىھتىياج ئۇچۇرلىرى بولۇپ بۇ ئۇچۇرلار ئارقىلىق شاڭخەيدىكى شىنجاڭلىق ئوقوغۇچىلارغائوقۇۋىتىپ خىزمەت قىلىپ ئائىلسىىگە بولغان ئىقسادى بىسىمنى يىنىكلىتىش پۇرسىتى بىلەن تەمىنلەيمىز،يەنە بىز ئۆزىمىز كۇتىۋالغان چەتئەللىك مۇسۇلمانلاردىن ساياھەت ئۆمىكى تەشكىللەپ شىنجاڭغا ساياھەتكە ئەۋەتىپ شىنجاڭ ساياھەتچىلىكىنىڭ گۇللىنىشى تۆھپە قوشىمىز،مۇسۇلمانلار مۇلازىمەت مەركىزىنىمۇ باشقا چوڭ شەھەرلەرگە كىڭەيتىش  پىلانىمىز بار ، بۇلاردىن باشقا بىرقانچە پىلانىمىز تىخى ئىش باشلىمىغان ھەم مەبلەخ سىلىپ ھەمكارلاشقۇچىلار بىلەن سۆھبەتلىشىۋاتقان بولغاچ بۇلار  سودا مەخپىيەتلىكىگە كىرىدۇ ، شۇڭا ۋاختىچە بۇلارنى سىزگە ئىيتىپ بىلەرمەيمەن .

سۇئال :سىزنىڭ ھازىرقى ئالى مەكتەپنى پۇتتۇرۇپ خىزمەت تاپالمايۋاتقان ، ۋە ئالى مەكتەپنى پۇتتۇرۇش ئالدىدا تۇرغان سىتودىنىتلارغا قانداق كۆڭۈل سۆزىڭىز بار؟
جاۋاپ: مىنىڭ ئەمدى ئالى مەكتەپ پۇتتۇرگەن ياكى پۇتتۇرۇش ئالدىدا تۇرىۋاتقان سىتودىنىتلارغا دەيدىغان كۆڭۈل سۆزلىرىم:
1: دۆلەت مەمۇرى بولۇش نۇۋەتتە ئالى مەكتەپ ئوقوغۇچىلىرىنىڭ بىردىن بىر چىقىش يولى ئەمەس.
ھەرقانداق بىر ئادەمنىڭ چوقۇم باشقىلار ئورنىنى باسالمايدىغان ئالاھىدىلىكى بولىدۇ، شۇنداق بولغانكەن ھازىرقى ئالى مەكتەپ ئوقۇغۇچىلىرى دۆلەت مەمۇرى بولۇشنى ئۆزىگە بىردىن بىر نىشان قىلىۋالماي ئۆزىدىكى ئاشۇ ئالاھىدىلىكنى بالدۇر بايقاپ ھەم شۇنداقلا ئۇنى جارى قىلدۇرۇپ جەمىيەتتىن ئۆزىنىڭ بىر كىشىلىك ئورنىنى تىپىشى كىرەك.
2: ھازىر شىنجاڭدىكى ئالى مەكتەپ ئوقوغۇچىلىرىدا ئەڭ ئىغىر بولغان مەسىلە،ئۆزىگە ئىشەنمەسلىك ، ئاتا-ئانىسىغا يۆلىنىۋىلىش، ئادەمدە يۇكسەك بىر ئىرادە ،ھەم شۇ ئىرادىگە لايىق تىرىشچانلىق بولىدىكەن ،بۇ دۇنيادا قىلغىلى بولمايدىغان ئىش يوق،شۇڭا مىنىڭ شىنجاڭدىكى ئالى مەكتەپ ئوقوغۇچىلىرىغا تەكلىۋىم: ئۆزەڭلا ئىشەنگەن ھالدا،ئۆزەڭلارنىڭ ئارزۇسى ئۈچۈن تىشچانلىق كۆرسۇتۇڭلا ، گۈزەل كەلگۈسى سىلەر ئىزدەنگۈچىلەرگە مەنسۈپ!!!!
مەن ئىگىلىك تىكلەش ئارزۇسى بار سىتۇدىنتلارنىڭ ھەر ۋاقىت مەن بىلەن ئالاقىلىشىپ مەسلىھەت سورىشىنى ھەم دائىم پىكىر ئالماشتۇرۇپ تۇرىشىنى ئۈمىت قىلىمەن ھەم چەتئەلگە ئوقۇشقا چىقىش ئارزۇسى بار سىتودىنىتلار مەن بىلەن ئالاقىلاشساڭلار بىجىرش قىيىن بولغان رەسمىيەتلەرنى( ئوقۇشقا قوبۇل قىلىش چاقىرقى، پاسپورت ، ۋىزاقاتارلىق)، بىجىرشىپ بىرىمەن .
