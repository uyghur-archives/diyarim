«ئۇنتۇش بىلەن لاتا»غا...


«ئاخىرقى تېمام»غا...

                         
                            نىمە بولدى بىلمىدىم، دىيارىمدا ھە بۇگۇن،
                            شائېرەنى، شائېرنى، تەتۇر شامال ئۇردىمۇ؟
                            ئەسەر يوللاپ بىر تاتلېق،كۆڭۈلنى غەش ئەيلىدى،
                            «ئاخىرقېسى» دەپ يىزىش،توردا ياكى مودىمۇ؟
                                  *            *              *              *
                            «ئۇنتۇش» دەپقۇ قويۇپسىز، ئىسمىڭىزنى چېرايلېق،
                            بىراق ئۇنتۇش مۇمكىنمۇ، دىيارىمدىن كەتسىڭىز؟
                            «ۋېسالسىز يار»نى تاپقاندا يا قايتىپ كىلەمسىز،
                            ھېجرانلارنى تۈگۈتۈپ، ۋېسالىغا يەتسىڭىز؟

                            كەتمەڭ ئۇنداق ئالدىراپ، تېخى مەشرەپ ئوينايمىز،
                            ئوتتۇز ئوغۇل، توققۈز قىز، دىيارىمدا جۆر بولۇپ.
                            نىسىپ بولسۇن كۆرۈشكە، يوللاپ تۇرۇڭ تېمىلار،
                            شېرىڭىزسىز قالمامدۇ، كۆزلىرىمىز كۆر بولۇپ.
                                 *          *             *              *
                            ساتېرانى ۋايىغا يەتكۈزگەنتى «لاتا»جان،
                            تەپپەككۈرى ئاقاتتى، دەريا كەبى لېقمۇ-لېق.
                            سىزسىز قانداق قېززىيدۇ مۇنبەردىكى شىئېريەت،
                            ئىسمىڭىزغا مۇناسىپ، قېلماڭ ئۇنداق لاتېلېق.

                           «قېسمىغىن» دەپ يىزىپسىز، شېئىرىڭنىغۇ بىر ئوبدان،
                            لىكىن ئاۋال ئۆزىڭىز، قېسقېنىڭىز نىمىسى؟
                            نىمە قېلاي دېدېڭىز، بۇ مۇنبەردىن ئايرىلىپ،
                            بولماقچىمۇ سودېگەر، ئېلىپ-سېتىپ نەق، نىسى؟

                            سىز كەتسىڭىز بولمايدۇ، قەلەم ئېلىڭ باشقىدىن،
                            سىزسىز بىزنى مۇنبەردە، بىر غىرىپلېق باسمامدۇ؟
                            تۇمەنلىگەن تورداشلار، ھوزۇر ئېلىپ شىئېردىن،
                            بوينىڭىزغا ھۆرمەتتە، گۈل-چەمبىرەك ئاسمامدۇ؟
